```
// assets/js/custody-por.js
// Custody & Proof-of-Reserves - Module 4 for RWA Scoring

const CUSTODY_DATA = {
  'BlackRock/Securitize': {
    custodian: 'Securitize + BNY Mellon',
    porStatus: 'Verified',
    porDate: '2026-03-15',
    reliabilityScore: 92,
    scoreBase: 830,
    metrics: { 'Insurance Coverage': 95, 'Audit Frequency': 90, 'Cold Storage %': 98, 'Redemption Reliability': 85 },
    attestationHistory: [820, 835, 840, 845, 830, 850, 830]
  },
  'Ondo Finance': {
    custodian: 'Coinbase Custody',
    porStatus: 'Verified',
    porDate: '2026-03-10',
    reliabilityScore: 88,
    scoreBase: 780,
    metrics: { 'Insurance Coverage': 90, 'Audit Frequency': 85, 'Cold Storage %': 92, 'Redemption Reliability': 88 },
    attestationHistory: [750, 760, 770, 775, 780]
  },
  'Centrifuge': {
    custodian: 'Fireblocks + Multiple',
    porStatus: 'Partially Verified',
    porDate: '2026-02-28',
    reliabilityScore: 65,
    scoreBase: 580,
    metrics: { 'Insurance Coverage': 70, 'Audit Frequency': 60, 'Cold Storage %': 75, 'Redemption Reliability': 55 },
    attestationHistory: [520, 550, 580, 590, 580]
  }
};

async function calculateCustodyScore(issuerName) {
  const data = CUSTODY_DATA[issuerName] || { scoreBase: 550, custodian: 'Unknown', porStatus: 'Unknown', reliabilityScore: 50, metrics: {}, attestationHistory: [500,510,520] };

  let score = data.scoreBase;
  score = Math.min(850, Math.max(300, Math.floor(score)));

  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 600 ? 'B' : score >= 500 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return {
    score,
    grade,
    tier,
    custodian: data.custodian,
    porStatus: data.porStatus,
    porDate: data.porDate,
    reliabilityScore: data.reliabilityScore,
    metrics: data.metrics,
    attestationHistory: data.attestationHistory
  };
}

async function renderCustodyPoR() {
  const container = document.getElementById('custody-por-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to verify custody & Proof-of-Reserves for your RWA holdings</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Verifying real-world asset backing and custodian Proof-of-Reserves...</p></div>`;

  try {
    const issuerName = "BlackRock/Securitize"; // In production, detect dynamically from wallet
    const result = await calculateCustodyScore(issuerName);

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Custody Score Card -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-cyan-600 mb-2">Custody & Proof-of-Reserves Score</h2>
          <div class="text-8xl font-bold text-cyan-600">${result.score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${result.grade}</div>
          <div class="inline-block px-6 py-2 bg-cyan-100 dark:bg-cyan-900 text-cyan-700 dark:text-cyan-200 rounded-3xl text-lg">${result.tier}</div>
          <p class="mt-8 text-sm text-gray-500">Custodian: ${result.custodian}</p>
        </div>

        <!-- PoR Status -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4 flex items-center gap-2">🔐 Proof-of-Reserves Status</h3>
          <div class="flex items-center gap-3 mb-6">
            <span class="px-4 py-2 rounded-3xl text-lg font-medium ${result.porStatus === 'Verified' ? 'bg-green-100 text-green-700' : 'bg-yellow-100 text-yellow-700'}">${result.porStatus}</span>
            <span class="text-sm text-gray-500">Last attested: ${result.porDate}</span>
          </div>
          <p class="text-sm">Reliability Score: <strong>${result.reliabilityScore}%</strong></p>
        </div>

        <!-- Custodian Reliability Metrics -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Custodian Reliability Metrics</h3>
          <canvas id="custodyMetricsChart" height="240"></canvas>
        </div>

        <!-- PoR Attestation History -->
        <div class="lg:col-span-3 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">PoR Attestation History (Last 7 attestations)</h3>
          <canvas id="porHistoryChart" height="180"></canvas>
        </div>
      </div>
    `;

    container.innerHTML = html;

    setTimeout(() => {
      new Chart(document.getElementById('custodyMetricsChart'), {
        type: 'bar',
        data: {
          labels: Object.keys(result.metrics),
          datasets: [{
            label: 'Score %',
            data: Object.values(result.metrics),
            backgroundColor: '#22d3ee',
            borderRadius: 8
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } }, scales: { y: { max: 100 } } }
      });

      new Chart(document.getElementById('porHistoryChart'), {
        type: 'line',
        data: {
          labels: ['Feb 10', 'Feb 18', 'Feb 25', 'Mar 5', 'Mar 12', 'Mar 20', 'Mar 28'],
          datasets: [{
            label: 'PoR Score',
            data: result.attestationHistory,
            borderColor: '#22d3ee',
            tension: 0.4,
            borderWidth: 3,
            pointRadius: 5
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } } }
      });
    }, 300);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not verify custody data. Please try again.</p>`;
  }
}

window.renderCustodyPoR = renderCustodyPoR;
```
