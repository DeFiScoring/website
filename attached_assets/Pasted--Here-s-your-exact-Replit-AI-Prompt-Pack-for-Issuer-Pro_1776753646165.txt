### Prompt 1: Create the Issuer Due Diligence JavaScript file
```
Create a new file exactly at assets/js/issuer-due-diligence.js with this complete, ready-to-use code:

// assets/js/issuer-due-diligence.js
// Issuer & Protocol Due Diligence - Module 2

const KNOWN_ISSUERS = {
  'BlackRock/Securitize': { scoreBase: 820, redFlags: 0, trend: 'Strong upward', defaults: 0, description: 'Institutional-grade, SEC-compliant, $10B+ AUM' },
  'Ondo Finance': { scoreBase: 780, redFlags: 1, trend: 'Rapid growth', defaults: 0, description: 'Leading tokenized Treasuries, strong partnerships' },
  'Centrifuge': { scoreBase: 640, redFlags: 2, trend: 'Moderate', defaults: 1, description: 'Private credit pioneer, some early default history' },
  'Maple Finance': { scoreBase: 520, redFlags: 3, trend: 'Volatile', defaults: 2, description: 'Institutional lending, past liquidity issues' },
};

async function calculateIssuerScore(issuerName) {
  const data = KNOWN_ISSUERS[issuerName] || { scoreBase: 600, redFlags: 2, trend: 'Neutral', defaults: 1, description: 'Unknown issuer' };

  let score = data.scoreBase;

  // Adjust based on red flags and defaults
  score -= data.redFlags * 45;
  score -= data.defaults * 60;
  score = Math.min(850, Math.max(300, Math.floor(score)));

  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 600 ? 'B' : score >= 500 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return {
    score,
    grade,
    tier,
    redFlags: data.redFlags,
    trend: data.trend,
    defaults: data.defaults,
    description: data.description
  };
}

async function renderIssuerDueDiligence() {
  const container = document.getElementById('issuer-diligence-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to detect RWA issuers and run due diligence</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Analyzing issuers behind your RWA holdings...</p></div>`;

  try {
    // For MVP, use the same top RWA asset from previous module or simulate
    const issuerName = "BlackRock/Securitize"; // In production, detect from wallet holdings
    const result = await calculateIssuerScore(issuerName);

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Issuer Score Card -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-violet-600 mb-2">Issuer Due Diligence Score</h2>
          <div class="text-8xl font-bold text-violet-600">${result.score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${result.grade}</div>
          <div class="inline-block px-6 py-2 bg-violet-100 dark:bg-violet-900 text-violet-700 dark:text-violet-200 rounded-3xl text-lg">${result.tier}</div>
          <p class="mt-8 text-sm text-gray-500">Issuer: ${issuerName}</p>
        </div>

        <!-- Red Flags & Summary -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4 flex items-center gap-2">🚩 Red Flag Summary</h3>
          <div class="space-y-4">
            <div class="flex justify-between"><span>Red Flags Detected</span><span class="font-medium text-red-600">${result.redFlags}</span></div>
            <div class="flex justify-between"><span>Historical Defaults</span><span class="font-medium">${result.defaults}</span></div>
            <div class="flex justify-between"><span>Performance Trend</span><span class="font-medium">${result.trend}</span></div>
            <p class="text-sm text-gray-600 dark:text-gray-400 mt-6">${result.description}</p>
          </div>
        </div>

        <!-- Trend Chart -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Issuer Performance Trend (12 months)</h3>
          <canvas id="issuerTrendChart" height="260"></canvas>
        </div>
      </div>
    `;

    container.innerHTML = html;

    // Render trend chart (fake historical data)
    setTimeout(() => {
      new Chart(document.getElementById('issuerTrendChart'), {
        type: 'line',
        data: {
          labels: ['Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec', 'Jan', 'Feb', 'Mar'],
          datasets: [{
            label: 'Issuer Score Trend',
            data: [720, 735, 748, 762, 755, 771, 780, 785, 790, 795, 810, result.score],
            borderColor: '#8b5cf6',
            tension: 0.4,
            borderWidth: 3,
            pointRadius: 4
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } } }
      });
    }, 300);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not analyze issuer data. Please try again.</p>`;
  }
}

// Make global
window.renderIssuerDueDiligence = renderIssuerDueDiligence;
```
