### Prompt 1: Create the approvals-checker.js file
```
Create a new file exactly at assets/js/approvals-checker.js with this complete, production-ready code for the Token Approvals Checker:

// assets/js/approvals-checker.js
// DeFi Token Approvals Checker - detects unlimited approvals on risky protocols

let ethersLib; // will hold ethers v6

const COMMON_TOKENS = {
  // Ethereum mainnet addresses (add more chains later)
  '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48': { symbol: 'USDC', name: 'USD Coin' },
  '0xdAC17F958D2ee523a2206206994597C13D831ec7': { symbol: 'USDT', name: 'Tether' },
  '0x6B175474E89094C44Da98b954EedeAC495271d0F': { symbol: 'DAI', name: 'Dai' },
  '0xC02aaA39b223FE8D0A0e5C4F27eAD9083C756Cc2': { symbol: 'WETH', name: 'Wrapped Ether' },
};

const RISKY_SPENDERS = {
  // Common high-risk / low-score protocols (you can expand this)
  '0x7a250d5630B4cF539739dF2C5dAcb4c659F2488D': { name: 'Uniswap V2 Router', risk: 'High' },
  '0x1111111254EEB25477B68fb85Ed929f73A960582': { name: '1inch Router', risk: 'Medium' },
  '0xC36442b4a4522E871399CD717aBDD847Ab11FE88': { name: 'Uniswap V3 Positions', risk: 'Low' },
  '0x68f180fcCe683185914c1f8f0C3A2cD4c8bD8d4B': { name: 'Aave V3 Pool (risky if low score)', risk: 'High' },
};

async function loadEthers() {
  if (ethersLib) return ethersLib;
  const script = document.createElement('script');
  script.src = 'https://cdn.jsdelivr.net/npm/ethers@6.13.0/dist/ethers.umd.min.js';
  await new Promise(resolve => { script.onload = resolve; document.head.appendChild(script); });
  ethersLib = window.ethers;
  return ethersLib;
}

async function checkTokenApprovals() {
  const container = document.getElementById('approvals-checker-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to scan token approvals</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Scanning approvals on Ethereum...</p></div>`;

  try {
    const ethers = await loadEthers();
    const provider = new ethers.BrowserProvider(window.ethereum);
    const signer = await provider.getSigner();
    const owner = await signer.getAddress();

    let rows = [];

    for (const [tokenAddr, tokenInfo] of Object.entries(COMMON_TOKENS)) {
      const tokenContract = new ethers.Contract(tokenAddr, [
        'function allowance(address owner, address spender) view returns (uint256)'
      ], provider);

      for (const [spenderAddr, spenderInfo] of Object.entries(RISKY_SPENDERS)) {
        const allowance = await tokenContract.allowance(owner, spenderAddr);

        const isUnlimited = allowance >= ethers.MaxUint256 / 2n; // practical "unlimited"

        if (allowance > 0n) {
          const riskLevel = isUnlimited ? '🔴 Unlimited' : '🟡 Limited';
          const protocolRisk = spenderInfo.risk === 'High' ? '🚨 Low-score protocol risk' : '';

          rows.push(`
            <tr class="border-b dark:border-gray-700 hover:bg-gray-50 dark:hover:bg-gray-800">
              <td class="px-6 py-4 font-medium">${tokenInfo.symbol}</td>
              <td class="px-6 py-4">${spenderInfo.name}</td>
              <td class="px-6 py-4 text-right font-mono">${isUnlimited ? '∞ Unlimited' : allowance.toString().slice(0,8)+'...'}</td>
              <td class="px-6 py-4"><span class="inline-block px-3 py-1 rounded-full text-xs ${isUnlimited ? 'bg-red-100 text-red-700' : 'bg-yellow-100 text-yellow-700'}">${riskLevel}</span></td>
              <td class="px-6 py-4">${protocolRisk}</td>
              <td class="px-6 py-4 text-right">
                <a href="https://revoke.cash/address/${owner}?chainId=1" target="_blank"
                   class="text-blue-600 hover:underline text-sm">Revoke →</a>
              </td>
            </tr>
          `);
        }
      }
    }

    const html = rows.length > 0 ? `
      <div class="overflow-x-auto">
        <table class="w-full text-sm">
          <thead>
            <tr class="border-b dark:border-gray-700">
              <th class="px-6 py-4 text-left">Token</th>
              <th class="px-6 py-4 text-left">Spender (Protocol)</th>
              <th class="px-6 py-4 text-right">Allowance</th>
              <th class="px-6 py-4">Risk</th>
              <th class="px-6 py-4">Note</th>
              <th class="px-6 py-4"></th>
            </tr>
          </thead>
          <tbody>${rows.join('')}</tbody>
        </table>
      </div>
      <p class="text-xs text-gray-500 mt-6">💡 Pro tip: Unlimited approvals on low-scoring protocols are the #1 cause of hacks. Revoke them immediately.</p>
    ` : `<div class="text-center py-12 text-green-600">✅ No risky approvals found on common tokens!</div>`;

    container.innerHTML = html;

  } catch (error) {
    console.error(error);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Error scanning approvals. Make sure wallet is connected and on Ethereum.</p>`;
  }
}

// Make it globally callable
window.checkTokenApprovals = checkTokenApprovals;
```
