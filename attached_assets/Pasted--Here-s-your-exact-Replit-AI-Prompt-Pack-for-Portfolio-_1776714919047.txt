### Prompt 1: Create the Portfolio RWA Exposure Aggregator JavaScript file
```
Create a new file exactly at assets/js/portfolio-rwa-exposure.js with this complete, ready-to-use code:

// assets/js/portfolio-rwa-exposure.js
// Portfolio RWA Exposure Aggregator - Module 8

const DEMO_RWA_HOLDINGS = [
  { asset: 'BUIDL', issuer: 'BlackRock/Securitize', value: 124000, riskScore: 820 },
  { asset: 'OUSG', issuer: 'Ondo Finance', value: 87000, riskScore: 780 },
  { asset: 'Centrifuge Pool', issuer: 'Centrifuge', value: 32000, riskScore: 580 }
];

async function calculatePortfolioScore(holdings) {
  if (!holdings || holdings.length === 0) return { score: 650, grade: 'B', tier: 'Moderate Risk', totalValue: 0, avgRisk: 0 };

  const totalValue = holdings.reduce((sum, h) => sum + h.value, 0);
  const weightedRisk = holdings.reduce((sum, h) => sum + (h.value * h.riskScore), 0) / totalValue;
  let score = Math.floor(weightedRisk);

  // Diversification bonus
  const diversificationBonus = Math.min(80, holdings.length * 25);
  score += diversificationBonus;

  score = Math.min(850, Math.max(300, score));

  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 600 ? 'B' : score >= 500 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return {
    score,
    grade,
    tier,
    totalValue,
    numAssets: holdings.length,
    avgRisk: Math.floor(weightedRisk),
    diversificationScore: diversificationBonus
  };
}

async function renderPortfolioRWAExposure() {
  const container = document.getElementById('portfolio-rwa-exposure-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to aggregate your full RWA portfolio exposure and cross-protocol risk</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Aggregating your RWA holdings across protocols...</p></div>`;

  try {
    const holdings = DEMO_RWA_HOLDINGS; // In production: scan wallet for RWA tokens
    const result = await calculatePortfolioScore(holdings);

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Portfolio Risk Score -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-purple-600 mb-2">Portfolio RWA Risk Score</h2>
          <div class="text-8xl font-bold text-purple-600">${result.score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${result.grade}</div>
          <div class="inline-block px-6 py-2 bg-purple-100 dark:bg-purple-900 text-purple-700 dark:text-purple-200 rounded-3xl text-lg">${result.tier}</div>
          <p class="mt-8 text-sm text-gray-500">Total RWA Exposure: <strong>$${(result.totalValue / 1000).toFixed(0)}k</strong></p>
        </div>

        <!-- Diversification Map -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Diversification Map</h3>
          <canvas id="diversificationChart" height="260"></canvas>
        </div>

        <!-- Portfolio Stats -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Portfolio Stats</h3>
          <div class="space-y-4 text-sm">
            <div class="flex justify-between"><span>Number of RWA Assets</span><span class="font-medium">${result.numAssets}</span></div>
            <div class="flex justify-between"><span>Average Asset Risk Score</span><span class="font-medium">${result.avgRisk}</span></div>
            <div class="flex justify-between"><span>Diversification Bonus</span><span class="font-medium text-purple-600">+${result.diversificationScore} points</span></div>
          </div>
        </div>

        <!-- Cross-Asset Correlation -->
        <div class="lg:col-span-3 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Cross-Asset Correlation Matrix</h3>
          <canvas id="correlationChart" height="220"></canvas>
        </div>
      </div>
    `;

    container.innerHTML = html;

    setTimeout(() => {
      // Diversification Pie Chart
      new Chart(document.getElementById('diversificationChart'), {
        type: 'pie',
        data: {
          labels: holdings.map(h => h.asset),
          datasets: [{
            data: holdings.map(h => h.value),
            backgroundColor: ['#a855f7', '#c026d3', '#7e22ce']
          }]
        },
        options: { responsive: true, plugins: { legend: { position: 'bottom' } } }
      });

      // Cross-Asset Correlation Heatmap (simple bar for MVP)
      new Chart(document.getElementById('correlationChart'), {
        type: 'bar',
        data: {
          labels: holdings.map(h => h.asset),
          datasets: [{
            label: 'Correlation to Portfolio',
            data: [0.12, 0.28, 0.65],
            backgroundColor: '#a855f7',
            borderRadius: 8
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } }, scales: { y: { max: 1 } } }
      });
    }, 300);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not aggregate portfolio data. Please try again.</p>`;
  }
}

window.renderPortfolioRWAExposure = renderPortfolioRWAExposure;
```
