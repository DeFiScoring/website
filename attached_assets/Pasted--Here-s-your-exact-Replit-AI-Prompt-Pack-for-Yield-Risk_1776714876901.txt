### Prompt 1: Create the Yield & Risk-Adjusted Performance JavaScript file
```
Create a new file exactly at assets/js/yield-risk-adjusted.js with this complete, ready-to-use code:

// assets/js/yield-risk-adjusted.js
// Yield & Risk-Adjusted Performance - Module 7 for RWA Scoring

const YIELD_DATA = {
  'BlackRock/Securitize': {
    yieldScoreBase: 810,
    annualYield: 4.8,
    sharpeRatio: 2.8,
    projectedDrawdown: 1.2,
    riskCorrelation: 0.15, // low correlation to crypto
    stressTestResults: [0.8, 1.1, 2.4, 3.8, 1.5] // 5 stress scenarios
  },
  'Ondo Finance': {
    yieldScoreBase: 770,
    annualYield: 5.2,
    sharpeRatio: 2.4,
    projectedDrawdown: 2.8,
    riskCorrelation: 0.35,
    stressTestResults: [1.5, 2.2, 4.1, 5.9, 2.8]
  },
  'Centrifuge': {
    yieldScoreBase: 520,
    annualYield: 8.5,
    sharpeRatio: 1.1,
    projectedDrawdown: 14.5,
    riskCorrelation: 0.75,
    stressTestResults: [6, 11, 19, 28, 9]
  }
};

async function calculateYieldScore(issuerName) {
  const data = YIELD_DATA[issuerName] || { yieldScoreBase: 550, annualYield: 3, sharpeRatio: 1.2, projectedDrawdown: 12, riskCorrelation: 0.6, stressTestResults: [5,8,15,20,7] };

  let score = data.yieldScoreBase;
  // Sharpe adjustment
  score += Math.floor(data.sharpeRatio * 35);
  // Penalty for high drawdown & correlation
  score -= Math.floor(data.projectedDrawdown * 4);
  if (data.riskCorrelation > 0.5) score -= 60;

  score = Math.min(850, Math.max(300, Math.floor(score)));

  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 600 ? 'B' : score >= 500 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return {
    score,
    grade,
    tier,
    annualYield: data.annualYield,
    sharpeRatio: data.sharpeRatio,
    projectedDrawdown: data.projectedDrawdown,
    riskCorrelation: data.riskCorrelation,
    stressTestResults: data.stressTestResults
  };
}

async function renderYieldRiskAdjusted() {
  const container = document.getElementById('yield-risk-adjusted-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to calculate risk-adjusted yield and stress-test scenarios for your RWA holdings</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Calculating Sharpe-like risk-adjusted yield and running stress-test scenarios...</p></div>`;

  try {
    const issuerName = "BlackRock/Securitize"; // In production, detect dynamically from wallet
    const result = await calculateYieldScore(issuerName);

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Yield Score Card -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-rose-600 mb-2">Yield & Risk-Adjusted Score</h2>
          <div class="text-8xl font-bold text-rose-600">${result.score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${result.grade}</div>
          <div class="inline-block px-6 py-2 bg-rose-100 dark:bg-rose-900 text-rose-700 dark:text-rose-200 rounded-3xl text-lg">${result.tier}</div>
          <p class="mt-8 text-sm text-gray-500">Annual Yield: <strong>${result.annualYield}%</strong></p>
        </div>

        <!-- Sharpe & Drawdown -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Risk-Adjusted Metrics</h3>
          <div class="space-y-5">
            <div class="flex justify-between"><span>Sharpe Ratio</span><span class="font-medium">${result.sharpeRatio}</span></div>
            <div class="flex justify-between"><span>Projected Max Drawdown</span><span class="font-medium text-red-600">${result.projectedDrawdown}%</span></div>
            <div class="flex justify-between"><span>Yield vs Risk Correlation</span><span class="font-medium">${result.riskCorrelation}</span></div>
          </div>
        </div>

        <!-- Stress Test Results -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Stress-Test Scenarios</h3>
          <canvas id="stressTestChart" height="240"></canvas>
        </div>

        <!-- Yield vs Risk Correlation -->
        <div class="lg:col-span-3 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Yield vs Risk Correlation (Historical)</h3>
          <canvas id="yieldRiskChart" height="180"></canvas>
        </div>
      </div>
    `;

    container.innerHTML = html;

    setTimeout(() => {
      // Stress Test Bar Chart
      new Chart(document.getElementById('stressTestChart'), {
        type: 'bar',
        data: {
          labels: ['Mild', 'Moderate', 'Severe', 'Extreme', 'Recovery'],
          datasets: [{
            label: 'Projected Drawdown %',
            data: result.stressTestResults,
            backgroundColor: '#f43f5e',
            borderRadius: 8
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } }, scales: { y: { beginAtZero: true } } }
      });

      // Yield vs Risk Correlation Line Chart
      new Chart(document.getElementById('yieldRiskChart'), {
        type: 'line',
        data: {
          labels: ['Jan', 'Feb', 'Mar', 'Apr', 'May'],
          datasets: [{
            label: 'Yield (%)',
            data: [4.2, 4.5, 4.8, 5.1, result.annualYield],
            borderColor: '#f43f5e',
            tension: 0.4,
            borderWidth: 3
          }, {
            label: 'Risk Correlation',
            data: [0.12, 0.18, 0.15, 0.22, result.riskCorrelation],
            borderColor: '#64748b',
            tension: 0.4,
            borderWidth: 3,
            yAxisID: 'y1'
          }]
        },
        options: { responsive: true, plugins: { legend: { display: true } }, scales: { y1: { position: 'right' } } }
      });
    }, 300);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not calculate yield data. Please try again.</p>`;
  }
}

window.renderYieldRiskAdjusted = renderYieldRiskAdjusted;
```
