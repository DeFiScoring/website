### Prompt 1: Create the dedicated RWA AI Chatbot Worker (new Repl)
```
Create a brand new Repl using the "Cloudflare Workers" template.

Name it exactly: DeFiScoring-rwa-chatbot

Replace src/index.js with this complete RWA-specific AI Worker code:

export default {
  async fetch(request, env) {
    if (request.method !== "POST") return new Response("Method not allowed", { status: 405 });

    const { action, sessionId, message, email } = await request.json();

    if (action === "submit-consent" && email) {
      await env.D1.prepare(`INSERT INTO leads (email, consented_at, source) VALUES (?, ?, 'rwa_chatbot') ON CONFLICT(email) DO UPDATE SET consented_at = ?`)
        .bind(email, new Date().toISOString(), new Date().toISOString()).run();
      return Response.json({ success: true });
    }

    if (!sessionId) return Response.json({ error: "sessionId required" }, { status: 400 });

    let history = await env.KV.get(`rwa-chat:${sessionId}`, { type: "json" }) || [];

    if (action === "send-message") {
      history.push({ role: "user", content: message });

      const aiResponse = await env.AI.run("@cf/meta/llama-3.1-8b-instruct-fast", {
        messages: [
          {
            role: "system",
            content: `You are an expert RWA Risk Advisor Chatbot. Ask 1 question at a time about the user's RWA holdings, goals, and risk tolerance.
            When the user says "finish", "report", or after 6-8 questions, output ONLY this exact JSON:
            {
              "riskProfile": "Conservative / Moderate / Aggressive",
              "summary": "2-3 paragraph personalized RWA summary",
              "recommendations": [{ "project": "...", "reason": "..." }],
              "overallPortfolioScore": 785,
              "pdfTitle": "Your Full RWA Institutional Risk Report"
            }`
          },
          ...history
        ],
        max_tokens: 1200,
        temperature: 0.7
      });

      const reply = aiResponse.response || aiResponse;
      history.push({ role: "assistant", content: reply });

      await env.KV.put(`rwa-chat:${sessionId}`, JSON.stringify(history), { expirationTtl: 3600 });

      let finalReport = null;
      try {
        if (reply.includes('{') && reply.includes('riskProfile')) {
          finalReport = JSON.parse(reply.match(/\{[\s\S]*\}/)[0]);
        }
      } catch (e) {}

      return Response.json({ success: true, reply, finalReport });
    }
    return Response.json({ error: "invalid action" }, { status: 400 });
  }
};
```

### Prompt 3: Create the frontend JS in your main Jekyll Repl
```
In the main Jekyll Repl, create assets/js/rwa-chatbot.js with this complete gated chat + PDF code:

// assets/js/rwa-chatbot.js
let rwaSessionId = 'rwa-session-' + Date.now();

async function startRWAChatbot() {
  const email = document.getElementById('rwa-lead-email').value.trim();
  if (!email) return alert("Please enter your email to unlock the RWA AI advisor");

  await fetch('https://defiscoring-rwa-chatbot.YOURUSERNAME.workers.dev', {
    method: 'POST',
    body: JSON.stringify({ action: 'submit-consent', email })
  });

  document.getElementById('rwa-consent-form').classList.add('hidden');
  document.getElementById('rwa-chat-container').classList.remove('hidden');

  addMessage('assistant', "Hi! I'm your dedicated RWA Risk Advisor. I'll ask targeted questions about your holdings, goals, and risk tolerance to build a full institutional report. Ready to begin?");
}

function addMessage(role, text) {
  const chat = document.getElementById('rwa-chat-messages');
  const div = document.createElement('div');
  div.className = `mb-4 ${role === 'user' ? 'text-right' : ''}`;
  div.innerHTML = `<div class="inline-block max-w-[80%] px-5 py-3 rounded-3xl ${role === 'user' ? 'bg-blue-600 text-white' : 'bg-gray-200 dark:bg-gray-800'}">${text}</div>`;
  chat.appendChild(div);
  chat.scrollTop = chat.scrollHeight;
}

async function sendRWAChatMessage() {
  const input = document.getElementById('rwa-chat-input');
  const msg = input.value.trim();
  if (!msg) return;
  addMessage('user', msg);
  input.value = '';

  const res = await fetch('https://defiscoring-rwa-chatbot.YOURUSERNAME.workers.dev', {
    method: 'POST',
    body: JSON.stringify({ action: 'send-message', sessionId: rwaSessionId, message: msg })
  });
  const data = await res.json();

  addMessage('assistant', data.reply);

  if (data.finalReport) {
    generateRWAPDF(data.finalReport);
  }
}

async function generateRWAPDF(report) {
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF();

  doc.setFontSize(20);
  doc.text(report.pdfTitle || "Your RWA Institutional Risk Report", 20, 30);
  doc.setFontSize(12);
  doc.text(`Risk Profile: ${report.riskProfile}`, 20, 50);
  doc.text(report.summary, 20, 65, { maxWidth: 160 });

  doc.save(`RWA-Personalized-Report-${new Date().toISOString().slice(0,10)}.pdf`);

  alert("✅ Beautiful PDF report downloaded! Thank you — we've added you to our institutional insights list.");
}

window.startRWAChatbot = startRWAChatbot;
window.sendRWAChatMessage = sendRWAChatMessage;
```
