### Prompt 1: Create the RWA score JavaScript file
```
Create a new file exactly at assets/js/rwa-asset-score.js with this complete, ready-to-use code:

// assets/js/rwa-asset-score.js
// RWA Asset Risk Score - Primary “FICO for RWAs”

const RWA_TOKENS = {
  // Common RWA token addresses (Ethereum + Base + others)
  '0x7712c34205737192402172409a8f7ccef8aa2aec': { symbol: 'BUIDL', name: 'BlackRock BUIDL', issuer: 'BlackRock/Securitize' },
  '0x6c3f90f043a72fa612cbac8115ee7e52bde6e490': { symbol: 'OUSG', name: 'Ondo OUSG', issuer: 'Ondo Finance' },
  // Add more as needed
};

async function calculateRWAScore(assetData, tvlCorrelation) {
  // Simple but transparent heuristic scoring (300-850)
  let score = 650; // base

  // TVL stability (higher & growing = better)
  if (tvlCorrelation > 1000000000) score += 120; // >$1B
  else if (tvlCorrelation > 500000000) score += 80;
  else if (tvlCorrelation > 100000000) score += 40;

  // Issuer reputation
  if (assetData.issuer.includes('BlackRock') || assetData.issuer.includes('Ondo')) score += 80;

  // Liquidity / redemption (from DeFiLlama category)
  if (assetData.category === 'Bond & MMF Funds') score += 60;
  if (assetData.redemptionRisk === 'low') score += 50;

  score = Math.min(850, Math.max(300, Math.floor(score)));
  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 650 ? 'B' : score >= 550 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return { score, grade, tier };
}

async function renderRWAScore() {
  const container = document.getElementById('rwa-score-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to scan RWA holdings and get your personalized RWA Asset Risk Score</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Fetching live RWA data from DeFiLlama...</p></div>`;

  try {
    // Fetch live RWA data from DeFiLlama
    const res = await fetch('https://api.llama.fi/protocols');
    const protocols = await res.json();
    const rwaProtocols = protocols.filter(p => p.category?.toLowerCase().includes('rwa') || p.name.toLowerCase().includes('buidl') || p.name.toLowerCase().includes('ondo'));

    // For MVP, use top RWA + wallet scan simulation
    const topAsset = rwaProtocols[0] || { name: 'BlackRock BUIDL', tvl: 3036000000, category: 'Bond & MMF Funds' };

    const tvlCorrelation = topAsset.tvl || 0;
    const assetData = { issuer: topAsset.name.includes('BlackRock') ? 'BlackRock/Securitize' : 'Ondo Finance', category: topAsset.category || 'Treasury' };

    const { score, grade, tier } = await calculateRWAScore(assetData, tvlCorrelation);

    // Breakdown data
    const breakdown = [
      { label: 'TVL Stability', value: Math.min(40, Math.floor(tvlCorrelation / 100000000)) },
      { label: 'Issuer Strength', value: 35 },
      { label: 'Liquidity & Redemption', value: 15 },
      { label: 'Oracle Integrity', value: 10 }
    ];

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Main RWA Score Card -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-emerald-600 mb-2">RWA Asset Risk Score</h2>
          <div class="text-8xl font-bold text-emerald-600">${score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${grade}</div>
          <div class="inline-block px-6 py-2 bg-emerald-100 dark:bg-emerald-900 text-emerald-700 dark:text-emerald-200 rounded-3xl text-lg">${tier}</div>
          <p class="mt-8 text-sm text-gray-500">Scanned from your wallet + live DeFiLlama data</p>
        </div>

        <!-- Breakdown -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Risk Breakdown</h3>
          <canvas id="rwaBreakdownChart" height="260"></canvas>
        </div>

        <!-- Top RWA Asset Info -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Top RWA Asset Detected</h3>
          <p class="text-2xl font-medium">${topAsset.name}</p>
          <p class="text-emerald-600">TVL: $${(topAsset.tvl / 1e9).toFixed(2)}B</p>
          <p class="text-sm mt-6">TVL Correlation: High stability (+${Math.floor(tvlCorrelation / 1e8)} points)</p>
        </div>
      </div>
    `;

    container.innerHTML = html;

    // Render breakdown chart
    setTimeout(() => {
      new Chart(document.getElementById('rwaBreakdownChart'), {
        type: 'bar',
        data: {
          labels: breakdown.map(b => b.label),
          datasets: [{ label: 'Weight %', data: breakdown.map(b => b.value), backgroundColor: '#10b981', borderRadius: 8 }]
        },
        options: { responsive: true, plugins: { legend: { display: false } } }
      });
    }, 200);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not fetch RWA data. Please try again.</p>`;
  }
}

// Make global
window.renderRWAScore = renderRWAScore;
```
