### Prompt 2: Replace the Worker code with the full AI Profiler
```
Replace the entire content of src/index.js with this complete, production-ready Cloudflare AI Worker code for the DeFi Risk Management Profiler:

export default {
  async fetch(request, env) {
    if (request.method !== "POST") {
      return new Response("Method not allowed", { status: 405 });
    }

    let body;
    try {
      body = await request.json();
    } catch (e) {
      return Response.json({ success: false, error: "Invalid JSON" }, { status: 400 });
    }

    const { wallet, deFiScore, recentActivity } = body;

    if (!wallet || !deFiScore) {
      return Response.json({ success: false, error: "wallet and deFiScore are required" }, { status: 400 });
    }

    // Call Cloudflare Workers AI (Llama 3.1 8B - fast version)
    const aiResponse = await env.AI.run("@cf/meta/llama-3.1-8b-instruct-fast", {
      messages: [
        {
          role: "system",
          content: `You are a senior DeFi risk advisor. Analyze the user's wallet data and give a concise risk profile + 3-5 tailored crypto project recommendations.
Output ONLY valid JSON in this exact format (no extra text):
{
  "riskProfile": "Conservative / Moderate / Aggressive",
  "summary": "short 2-3 sentence summary",
  "recommendations": [
    { "project": "project name", "reason": "why it fits this user", "riskLevel": "Low/Medium/High" }
  ]
}`
        },
        {
          role: "user",
          content: `Wallet address: ${wallet}
DeFi Credit Score: ${deFiScore}/850
Recent activity summary: ${recentActivity || "No activity provided"}`
        }
      ],
      max_tokens: 800,
      temperature: 0.7
    });

    let profile;
    try {
      profile = JSON.parse(aiResponse.response || aiResponse);
    } catch (e) {
      profile = { riskProfile: "Moderate", summary: "AI parsing error - please try again", recommendations: [] };
    }

    return Response.json({
      success: true,
      profile,
      timestamp: new Date().toISOString()
    });
  }
};
```
