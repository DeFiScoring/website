### Prompt 2: Replace the Worker code (secure GitHub API proxy)
```
Replace the entire content of src/index.js with this complete Worker code:

export default {
  async fetch(request, env) {
    if (request.method !== "POST") {
      return new Response("Method not allowed", { status: 405 });
    }

    let data;
    try {
      data = await request.json();
    } catch (e) {
      return Response.json({ success: false, error: "Invalid JSON" }, { status: 400 });
    }

    const { title, body, labels = ["user-report"] } = data;

    if (!title || !body) {
      return Response.json({ success: false, error: "title and body required" }, { status: 400 });
    }

    const githubResponse = await fetch("https://api.github.com/repos/DeFiScoring/website/issues", {
      method: "POST",
      headers: {
        "Authorization": `token ${env.GITHUB_TOKEN}`,
        "Accept": "application/vnd.github.v3+json",
        "Content-Type": "application/json",
        "User-Agent": "DeFiScoring-Report-Issue-Worker"
      },
      body: JSON.stringify({
        title: title,
        body: body,
        labels: labels
      })
    });

    if (!githubResponse.ok) {
      const errorText = await githubResponse.text();
      return Response.json({ success: false, error: `GitHub API error: ${githubResponse.status} ${errorText}` }, { status: 502 });
    }

    const issue = await githubResponse.json();

    return Response.json({
      success: true,
      issueUrl: issue.html_url,
      issueNumber: issue.number
    });
  }
};
```

### Prompt 5: Add the Report Issue form + JS to your main Jekyll site (in main Repl `DeFiScoring/website`)
```
Now switch to the main Jekyll Repl (DeFiScoring/website).

Create a new file at assets/js/report-issue.js with this code:

// assets/js/report-issue.js
async function submitReportIssue() {
  const form = document.getElementById('report-issue-form');
  const button = document.getElementById('report-button');
  const result = document.getElementById('report-result');

  const title = document.getElementById('issue-title').value.trim();
  const description = document.getElementById('issue-description').value.trim();

  if (!title || !description) {
    result.innerHTML = `<p class="text-red-500">Please fill in both title and description.</p>`;
    return;
  }

  button.disabled = true;
  button.innerHTML = 'Creating GitHub Issue...';

  try {
    const res = await fetch('https://defiscoring-report-issue.YOURUSERNAME.workers.dev', {  // ← replace with real Worker URL after deploy
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        title: title,
        body: `${description}\n\n---\nReported from: ${window.location.href}\nWallet: ${window.userWallet || 'Not connected'}\nUser Agent: ${navigator.userAgent}`,
        labels: ['user-report', 'dashboard']
      })
    });

    const data = await res.json();

    if (data.success) {
      result.innerHTML = `
        <div class="bg-green-100 dark:bg-green-900 p-6 rounded-3xl">
          <p class="font-medium">✅ Issue #${data.issueNumber} created successfully!</p>
          <a href="${data.issueUrl}" target="_blank" class="text-blue-600 hover:underline block mt-2">View on GitHub →</a>
        </div>`;
      form.reset();
    } else {
      result.innerHTML = `<p class="text-red-500">Error: ${data.error}</p>`;
    }
  } catch (e) {
    result.innerHTML = `<p class="text-red-500">Connection error. Please try again.</p>`;
  } finally {
    button.disabled = false;
    button.innerHTML = 'Submit to GitHub';
  }
}

// Make global
window.submitReportIssue = submitReportIssue;
```
