### Prompt 1: Create the Legal & Regulatory Compliance JavaScript file
```
Create a new file exactly at assets/js/legal-compliance.js with this complete, ready-to-use code:

// assets/js/legal-compliance.js
// Legal & Regulatory Compliance - Module 3 for RWA Scoring

const REGULATORY_DATA = {
  'BlackRock/Securitize': {
    jurisdiction: 'USA',
    kycAml: 'High',
    micaSec: 'SEC-registered + MiCA-ready',
    transferRestrictions: 'Moderate (whitelist only)',
    complianceScoreBase: 810,
    riskMap: { USA: 'Low', EU: 'Low', Global: 'Medium' },
    heatmapData: [95, 88, 92, 85], // KYC, AML, SEC, MiCA
    trendData: [720, 740, 765, 790, 805, 810, 815, 820]
  },
  'Ondo Finance': {
    jurisdiction: 'USA / Cayman',
    kycAml: 'High',
    micaSec: 'SEC-compliant',
    transferRestrictions: 'Low',
    complianceScoreBase: 760,
    riskMap: { USA: 'Low', EU: 'Medium', Global: 'Low' },
    heatmapData: [90, 85, 88, 78],
    trendData: [680, 700, 720, 745, 755, 760]
  },
  'Centrifuge': {
    jurisdiction: 'EU / Global',
    kycAml: 'Medium',
    micaSec: 'MiCA-aligned',
    transferRestrictions: 'High',
    complianceScoreBase: 620,
    riskMap: { USA: 'High', EU: 'Low', Global: 'Medium' },
    heatmapData: [75, 70, 65, 82],
    trendData: [580, 600, 610, 625, 630]
  }
};

async function calculateComplianceScore(issuerName) {
  const data = REGULATORY_DATA[issuerName] || { complianceScoreBase: 550, jurisdiction: 'Unknown', kycAml: 'Unknown', micaSec: 'Unknown', transferRestrictions: 'Unknown', riskMap: {}, heatmapData: [50,50,50,50], trendData: [500,520,530,540] };

  let score = data.complianceScoreBase;
  score = Math.min(850, Math.max(300, Math.floor(score)));

  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 600 ? 'B' : score >= 500 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return {
    score,
    grade,
    tier,
    jurisdiction: data.jurisdiction,
    kycAml: data.kycAml,
    micaSec: data.micaSec,
    transferRestrictions: data.transferRestrictions,
    riskMap: data.riskMap,
    heatmapData: data.heatmapData,
    trendData: data.trendData
  };
}

async function renderLegalCompliance() {
  const container = document.getElementById('legal-compliance-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to run legal & regulatory compliance check on detected RWA issuers</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Analyzing regulatory status for detected RWA issuers...</p></div>`;

  try {
    // Use same issuer as previous module for continuity (in production detect dynamically)
    const issuerName = "BlackRock/Securitize";
    const result = await calculateComplianceScore(issuerName);

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Compliance Score Card -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-amber-600 mb-2">Legal & Regulatory Compliance Grade</h2>
          <div class="text-8xl font-bold text-amber-600">${result.score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${result.grade}</div>
          <div class="inline-block px-6 py-2 bg-amber-100 dark:bg-amber-900 text-amber-700 dark:text-amber-200 rounded-3xl text-lg">${result.tier}</div>
          <p class="mt-8 text-sm text-gray-500">Issuer: ${issuerName}</p>
        </div>

        <!-- Jurisdiction Risk Map -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4 flex items-center gap-2">🌍 Jurisdiction Risk Map</h3>
          <div class="grid grid-cols-3 gap-4 text-center">
            ${Object.entries(result.riskMap).map(([jur, risk]) => `
              <div class="p-4 rounded-2xl ${risk === 'Low' ? 'bg-green-100 text-green-700' : risk === 'Medium' ? 'bg-yellow-100 text-yellow-700' : 'bg-red-100 text-red-700'}">
                <div class="font-medium">${jur}</div>
                <div class="text-2xl">${risk}</div>
              </div>
            `).join('')}
          </div>
          <div class="mt-6 text-xs text-gray-500">KYC/AML • MiCA/SEC • Transfer Restrictions</div>
        </div>

        <!-- Regulatory Exposure Heatmap -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Regulatory Exposure Heatmap</h3>
          <canvas id="complianceHeatmap" height="240"></canvas>
        </div>

        <!-- Compliance Trends -->
        <div class="lg:col-span-3 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Compliance Trend (Last 8 Months)</h3>
          <canvas id="complianceTrendChart" height="180"></canvas>
        </div>
      </div>
    `;

    container.innerHTML = html;

    // Render Heatmap (bar chart for simplicity)
    setTimeout(() => {
      new Chart(document.getElementById('complianceHeatmap'), {
        type: 'bar',
        data: {
          labels: ['KYC/AML', 'SEC', 'MiCA', 'Transfer Restrictions'],
          datasets: [{
            label: 'Compliance Strength %',
            data: result.heatmapData,
            backgroundColor: ['#10b981', '#8b5cf6', '#f59e0b', '#ef4444'],
            borderRadius: 8
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } }, scales: { y: { max: 100 } } }
      });

      // Render Trend Chart
      new Chart(document.getElementById('complianceTrendChart'), {
        type: 'line',
        data: {
          labels: ['Aug', 'Sep', 'Oct', 'Nov', 'Dec', 'Jan', 'Feb', 'Mar'],
          datasets: [{
            label: 'Compliance Score',
            data: result.trendData,
            borderColor: '#f59e0b',
            tension: 0.3,
            borderWidth: 3,
            pointRadius: 5
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } } }
      });
    }, 300);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not fetch regulatory data. Please try again.</p>`;
  }
}

// Make global
window.renderLegalCompliance = renderLegalCompliance;
```
