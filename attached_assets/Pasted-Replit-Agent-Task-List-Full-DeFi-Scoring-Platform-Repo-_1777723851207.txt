// worker/lib/providers.js
// Unified read interface across Alchemy / Moralis / Covalent.
// Each function takes (chain, env, ...) and returns normalized data.

import { CHAINS_BY_ID } from './chains.js';

// ----- Alchemy ------------------------------------------------------------
async function alchemyRpc(chain, env, method, params) {
  if (!chain.alchemy) throw new Error(`no alchemy for ${chain.id}`);
  const url = `https://${chain.alchemy}.g.alchemy.com/v2/${env.ALCHEMY_KEY}`;
  const r = await fetch(url, {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({ jsonrpc: '2.0', id: 1, method, params }),
  });
  const j = await r.json();
  if (j.error) throw new Error(`alchemy:${method}:${j.error.message}`);
  return j.result;
}

export async function getNativeBalance(chain, env, address) {
  if (chain.alchemy) {
    const hex = await alchemyRpc(chain, env, 'eth_getBalance', [address, 'latest']);
    return Number(BigInt(hex)) / 1e18;
  }
  // Moralis fallback
  return moralisNativeBalance(chain, env, address);
}

export async function getErc20Balances(chain, env, address) {
  if (chain.alchemy) {
    const res = await alchemyRpc(chain, env, 'alchemy_getTokenBalances', [address, 'erc20']);
    const filtered = (res.tokenBalances || []).filter((t) => t.tokenBalance && t.tokenBalance !== '0x0');
    const enriched = await Promise.all(filtered.map(async (t) => {
      const meta = await alchemyRpc(chain, env, 'alchemy_getTokenMetadata', [t.contractAddress]).catch(() => ({}));
      const decimals = meta.decimals ?? 18;
      const amount = Number(BigInt(t.tokenBalance)) / Math.pow(10, decimals);
      return {
        contract: t.contractAddress.toLowerCase(),
        symbol: meta.symbol || 'UNKNOWN',
        name: meta.name || 'Unknown',
        logo: meta.logo || null,
        decimals,
        amount,
      };
    }));
    return enriched.filter((t) => t.amount > 0);
  }
  return moralisErc20Balances(chain, env, address);
}

export async function getNfts(chain, env, address) {
  if (chain.alchemy) {
    const url = `https://${chain.alchemy}.g.alchemy.com/nft/v3/${env.ALCHEMY_KEY}/getNFTsForOwner?owner=${address}&pageSize=100&withMetadata=true`;
    const r = await fetch(url);
    if (!r.ok) return [];
    const j = await r.json();
    return (j.ownedNfts || []).map((n) => ({
      chain: chain.id,
      contract: n.contract?.address?.toLowerCase(),
      collection: n.contract?.name || n.contract?.openSeaMetadata?.collectionName || 'Unknown',
      tokenId: n.tokenId,
      title: n.name || `#${n.tokenId}`,
      image: n.image?.cachedUrl || n.image?.thumbnailUrl || n.image?.originalUrl || null,
      floorEth: n.contract?.openSeaMetadata?.floorPrice ?? null,
      tokenStandard: n.tokenType,
    }));
  }
  return moralisNfts(chain, env, address);
}

export async function getTransactionCount(chain, env, address) {
  if (chain.alchemy) {
    const hex = await alchemyRpc(chain, env, 'eth_getTransactionCount', [address, 'latest']);
    return Number(BigInt(hex));
  }
  return 0;
}

async function moralisNativeBalance(chain, env, address) {
  if (!env.MORALIS_KEY) return 0;
  const url = `https://deep-index.moralis.io/api/v2.2/${address}/balance?chain=${chain.moralis}`;
  const r = await fetch(url, { headers: { 'X-API-Key': env.MORALIS_KEY } });
  if (!r.ok) return 0;
  const j = await r.json();
  return Number(j.balance || 0) / 1e18;
}

async function moralisErc20Balances(chain, env, address) {
  if (!env.MORALIS_KEY) return [];

💰 Task 4 — Price Layer with Caching
Goal: Batch price fetching for tokens + native assets, with KV cache to stay under free-tier limits.
Do:
Create worker/lib/cache.js:
javascript
// worker/lib/cache.js
// Thin wrapper over Cloudflare KV. Falls back to in-memory Map in dev.

const memoryCache = new Map();

export async function cacheGet(env, key) {
  if (env.CACHE) {
    const v = await env.CACHE.get(key);
    return v ? JSON.parse(v) : null;
  }
  const entry = memoryCache.get(key);
  if (!entry) return null;
  if (entry.expires < Date.now()) { memoryCache.delete(key); return null; }
  return entry.value;
}

export async function cacheSet(env, key, value, ttlSeconds = 60) {
  if (env.CACHE) {
    await env.CACHE.put(key, JSON.stringify(value), { expirationTtl: ttlSeconds });
    return;
  }
  memoryCache.set(key, { value, expires: Date.now() + ttlSeconds * 1000 });
}
Create worker/lib/prices.js:
javascript
// worker/lib/prices.js
import { cacheGet, cacheSet } from './cache.js';

const CG_FREE = 'https://api.coingecko.com/api/v3';
const CG_PRO = 'https://pro-api.coingecko.com/api/v3';

function cgBase(env) { return env.COINGECKO_KEY ? CG_PRO : CG_FREE; }
function cgHeaders(env) { return env.COINGECKO_KEY ? { 'x-cg-pro-api-key': env.COINGECKO_KEY } : {}; }

export async function priceTokens(chain, env, contracts, fiat) {
  if (!contracts.length) return {};
  const fiatLow = fiat.toLowerCase();
  const cacheKey = `px:${chain.id}:${fiatLow}:${contracts.sort().join(',')}`;
  const cached = await cacheGet(env, cacheKey);
  if (cached) return cached;

  const url = `${cgBase(env)}/simple/token_price/${chain.coingecko}?contract_addresses=${contracts.join(',')}&vs_currencies=${fiatLow}`;
  const r = await fetch(url, { headers: cgHeaders(env) });
  if (!r.ok) return {};
  const j = await r.json();
  await cacheSet(env, cacheKey, j, 60);
  return j;
}

export async function priceNative(chain, env, fiat) {
  const fiatLow = fiat.toLowerCase();
  const cacheKey = `px:native:${chain.nativeCoingeckoId}:${fiatLow}`;
  const cached = await cacheGet(env, cacheKey);
  if (cached !== null) return cached;

  const url = `${cgBase(env)}/simple/price?ids=${chain.nativeCoingeckoId}&vs_currencies=${fiatLow}`;
  const r = await fetch(url, { headers: cgHeaders(env) });
  if (!r.ok) return 0;
  const j = await r.json();
  const price = j[chain.nativeCoingeckoId]?.[fiatLow] ?? 0;
  await cacheSet(env, cacheKey, price, 60);
  return price;
}

export async function priceMultipleNatives(env, fiat) {
  const fiatLow = fiat.toLowerCase();
  const ids = ['ethereum', 'polygon-ecosystem-token', 'binancecoin', 'avalanche-2', 'xdai'];
  const cacheKey = `px:natives:${fiatLow}`;
  const cached = await cacheGet(env, cacheKey);
  if (cached) return cached;

  const url = `${cgBase(env)}/simple/price?ids=${ids.join(',')}&vs_currencies=${fiatLow}`;
  const r = await fetch(url, { headers: cgHeaders(env) });
  if (!r.ok) return {};
  const j = await r.json();
  await cacheSet(env, cacheKey, j, 60);
  return j;
}

// worker/handlers/portfolio.js
import { CHAINS, CHAINS_BY_ID } from '../lib/chains.js';
import { getNativeBalance, getErc20Balances } from '../lib/providers.js';
import { priceTokens, priceNative } from '../lib/prices.js';

const isAddress = (a) => /^0x[a-fA-F0-9]{40}$/.test(a || '');

async function scanChain(chain, env, address, fiat) {
  const [native, tokens] = await Promise.all([
    getNativeBalance(chain, env, address).catch(() => 0),
    getErc20Balances(chain, env, address).catch(() => []),
  ]);

  const [nativePx, tokenPxs] = await Promise.all([
    priceNative(chain, env, fiat).catch(() => 0),
    priceTokens(chain, env, tokens.map((t) => t.contract), fiat).catch(() => ({})),
  ]);

  const tokenRows = tokens.map((t) => {
    const px = tokenPxs[t.contract.toLowerCase()]?.[fiat.toLowerCase()] ?? 0;
    return { ...t, chain: chain.id, priceFiat: px, valueFiat: px * t.amount };
  });

  const nativeRow = native > 0 ? [{
    contract: 'native',
    symbol: chain.nativeSymbol,
    name: `${chain.name} native`,
    chain: chain.id,
    decimals: 18,
    amount: native,
    priceFiat: nativePx,
    valueFiat: nativePx * native,
    logo: null,
  }] : [];

  const allTokens = [...nativeRow, ...tokenRows];
  return {
    chain: chain.id,
    chainName: chain.name,
    chainId: chain.chainId,
    tier: chain.tier,
    tokens: allTokens.sort((a, b) => b.valueFiat - a.valueFiat),
    totalFiat: allTokens.reduce((s, t) => s + (t.valueFiat || 0), 0),
  };
}

export async function handlePortfolio(request, env) {
  const url = new URL(request.url);
  const address = (url.searchParams.get('address') || '').toLowerCase();
  const fiat = (url.searchParams.get('fiat') || 'USD').toUpperCase();
  const chainFilter = url.searchParams.get('chains'); // optional CSV

  if (!isAddress(address)) {
    return json({ error: 'invalid address' }, 400);
  }

  const chainsToScan = chainFilter
    ? CHAINS.filter((c) => chainFilter.split(',').includes(c.id))
    : CHAINS;

  const perChain = await Promise.all(chainsToScan.map((c) =>
    scanChain(c, env, address, fiat).catch((err) => ({
      chain: c.id, chainName: c.name, chainId: c.chainId, tier: c.tier,
      tokens: [], totalFiat: 0, error: String(err),
    }))
  ));

  const portfolioFiat = perChain.reduce((s, c) => s + c.totalFiat, 0);
  const activeChains = perChain.filter((c) => c.totalFiat > 0).length;
  const totalTokens = perChain.reduce((s, c) => s + c.tokens.length, 0);

  return json({
    address,
    fiat,
    portfolioFiat,
    activeChains,
    totalTokens,
    chains: perChain,
    updatedAt: new Date().toISOString(),
  });
}

function json(data, status = 200) {
  return new Response(JSON.stringify(data, null, 2), {
    status,
    headers: {
      'content-type': 'application/json',
      'cache-control': 'public, max-age=30',
      'access-control-allow-origin': '*',
    },
  });
}

// getUserAccountData(address) returns:
//   totalCollateralBase, totalDebtBase, availableBorrowsBase,
//   currentLiquidationThreshold, ltv, healthFactor
const GET_USER_ACCOUNT_DATA = '0xbf92857c'; // function selector

async function readAaveV3(chain, env, address) {
  const pool = AAVE_V3_POOL[chain.id];
  if (!pool || !chain.alchemy) return null;
  const url = `https://${chain.alchemy}.g.alchemy.com/v2/${env.ALCHEMY_KEY}`;
  const data = GET_USER_ACCOUNT_DATA + address.replace(/^0x/, '').padStart(64, '0');
  const r = await fetch(url, {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({
      jsonrpc: '2.0', id: 1, method: 'eth_call',
      params: [{ to: pool, data }, 'latest'],
    }),
  });
  const j = await r.json();
  if (!j.result || j.result === '0x') return null;
  const hex = j.result.replace(/^0x/, '');
  if (hex.length < 64 * 6) return null;
  const parse = (i) => BigInt('0x' + hex.slice(i * 64, (i + 1) * 64));
  const totalCollateral = Number(parse(0)) / 1e8; // USD with 8 decimals
  const totalDebt = Number(parse(1)) / 1e8;
  const ltv = Number(parse(3)) / 10000;
  const healthFactor = Number(parse(5)) / 1e18;
  if (totalCollateral === 0 && totalDebt === 0) return null;
  return {

// worker/scoring/factors.js
// Each factor returns 0..maxWeight. Sum is mapped to FICO-style band.

export function onChainActivityFactor(portfolio) {
  const tokenCount = portfolio.totalTokens || 0;
  return Math.min(20, tokenCount * 1.2);
}

export function multiChainDiversityFactor(portfolio) {
  const active = portfolio.activeChains || 0;
  return Math.min(15, active * 3);
}

export function nativeLiquidityFactor(portfolio) {
  const v = portfolio.portfolioFiat || 0;
  if (v >= 10000) return 20;
  if (v >= 1000) return 15;
  if (v >= 100) return 8;
  if (v >= 10) return 3;
  return 1;
}

export function walletAgeFactor(firstTxTimestamp) {
  if (!firstTxTimestamp) return 5;
  const days = (Date.now() - firstTxTimestamp) / (1000 * 60 * 60 * 24);
  if (days >= 1095) return 15; // 3+ years
  if (days >= 365) return 12;
  if (days >= 180) return 8;
  if (days >= 30) return 4;
  return 1;
}

export function contractDiversityFactor(portfolio) {
  const contracts = new Set();
  for (const c of portfolio.chains || []) {
    for (const t of c.tokens || []) {
      if (t.contract !== 'native') contracts.add(t.contract);
    }
  }
  return Math.min(15, contracts.size * 0.8);
}

export function tokenDiversityFactor(portfolio) {
  const symbols = new Set();
  for (const c of portfolio.chains || []) {
    for (const t of c.tokens || []) symbols.add(t.symbol);
  }
  return Math.min(10, symbols.size * 0.7);
}

export function defiEngagementFactor(positions) {
  const count = positions.positionCount || 0;
  const atRisk = positions.atRiskCount || 0;
  const base = Math.min(10, count * 2);
  return Math.max(0, base - atRisk * 3);
}

export function nftHoldingsFactor(nftData) {
  const v = nftData.totalEstimatedValueFiat || 0;
  if (v >= 50000) return 5;
  if (v >= 5000) return 4;
  if (v >= 500) return 3;
  if (v >= 50) return 2;
  if ((nftData.totalNfts || 0) > 0) return 1;
  return 0;
}

export function toFicoScore(raw) {
  const clamped = Math.max(0, Math.min(100, raw));
  return Math.round(300 + (clamped / 100) * 550);

// Generates real-time alerts. For now, computes from current state;

const DUST_THRESHOLD_USD = 0.01;

// Hardcoded suspicious-token denylist; in production, sync from a public list.
const SUSPICIOUS_PATTERNS = [
  /visit.*claim/i,
  /\.com$/i,
  /\.io$/i,
  /airdrop/i,
  /reward/i,
];

export async function handleAlerts(request, env) {
  const url = new URL(request.url);
  const address = (url.searchParams.get('address') || '').toLowerCase();
  if (!isAddress(address)) {
    return new Response(JSON.stringify({ error: 'invalid address' }), {
      status: 400, headers: { 'content-type': 'application/json' },
    });
  }

  const baseUrl = url.origin;
  const [portRes, posRes] = await Promise.all([
    handlePortfolio(new Request(`${baseUrl}/api/portfolio?address=${address}&fiat=USD`), env),
    handleDefiPositions(new Request(`${baseUrl}/api/defi-positions?address=${address}`), env),
  ]);
  const [portfolio, positions] = await Promise.all([portRes.json(), posRes.json()]);

  const alerts = [];

  // 1. Health-factor alerts
  for (const p of (positions.positions || [])) {
    if (p.healthFactor != null && p.healthFactor < 1.2) {
      alerts.push({
        severity: 'critical',
        type: 'liquidation_risk',
        title: `${p.protocol} position near liquidation`,
        message: `Health factor: ${p.healthFactor.toFixed(2)} on ${p.chain}. Add collateral or repay.`,
        actionUrl: `https://app.aave.com`,
      });
    } else if (p.healthFactor != null && p.healthFactor < 1.5) {
      alerts.push({
        severity: 'warning',
        type: 'low_health',
        title: `${p.protocol} health factor low`,
        message: `Health factor: ${p.healthFactor.toFixed(2)} on ${p.chain}.`,
      });
    }
  }

  // 2. Suspicious tokens (likely scam airdrops)
  for (const c of (portfolio.chains || [])) {
    for (const t of (c.tokens || [])) {
      const looksScam = SUSPICIOUS_PATTERNS.some((re) => re.test(t.name || '') || re.test(t.symbol || ''));
      if (looksScam) {
        alerts.push({
          severity: 'warning',
          type: 'suspicious_token',
          title: `Suspicious token detected: ${t.symbol}`,
          message: `On ${c.chainName}. Do NOT interact — likely scam airdrop. Hide or revoke approvals.`,
          tokenContract: t.contract,
          chain: c.chain,
        });
      }
    }
  }

  // 3. Dust (informational)
  let dustCount = 0;
  for (const c of (portfolio.chains || [])) {
    for (const t of (c.tokens || [])) {
      if (t.valueFiat > 0 && t.valueFiat < DUST_THRESHOLD_USD) dustCount++;
    }
  }
  if (dustCount >= 5) {
    alerts.push({
      severity: 'info',
      type: 'dust',
      title: `${dustCount} dust tokens detected`,
      message: 'Consider sweeping to consolidate.',
    });
  }

  // 4. Single-chain concentration
  const sorted = (portfolio.chains || []).slice().sort((a, b) => b.totalFiat - a.totalFiat);
  if (sorted[0]?.totalFiat > 0 && sorted[0].totalFiat / portfolio.portfolioFiat > 0.9) {
    alerts.push({
      severity: 'info',
      type: 'concentration',
      title: 'Portfolio concentrated on one chain',
      message: `${(sorted[0].totalFiat / portfolio.portfolioFiat * 100).toFixed(0)}% on ${sorted[0].chainName}.`,
    });
  }

  return new Response(JSON.stringify({
    address,
    alertCount: alerts.length,
    criticalCount: alerts.filter((a) => a.severity === 'critical').length,
    alerts,
    updatedAt: new Date().toISOString(),
  }, null, 2), {
    headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },
  });
}

          headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },

async function alchemyRpc(chain, env, method, params) {
  if (!chain.alchemy) throw new Error(`no alchemy for ${chain.id}`);
  const url = `https://${chain.alchemy}.g.alchemy.com/v2/${env.ALCHEMY_KEY}`;
  const r = await fetch(url, {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({ jsonrpc: '2.0', id: 1, method, params }),
  });
  const j = await r.json();
  if (j.error) throw new Error(`alchemy:${method}:${j.error.message}`);
  return j.result;
}

💰 Task 4 — Price Layer with Caching
Goal: Batch price fetching for tokens + native assets, with KV cache to stay under free-tier limits.

// worker/lib/cache.js
// Thin wrapper over Cloudflare KV. Falls back to in-memory Map in dev.

const memoryCache = new Map();

export async function cacheGet(env, key) {
  if (env.CACHE) {
    const v = await env.CACHE.get(key);
    return v ? JSON.parse(v) : null;
  }
  const entry = memoryCache.get(key);
  if (!entry) return null;
  if (entry.expires < Date.now()) { memoryCache.delete(key); return null; }
  return entry.value;
}

export async function cacheSet(env, key, value, ttlSeconds = 60) {
  if (env.CACHE) {
    await env.CACHE.put(key, JSON.stringify(value), { expirationTtl: ttlSeconds });
    return;
  }
  memoryCache.set(key, { value, expires: Date.now() + ttlSeconds * 1000 });
}

const CG_PRO = 'https://pro-api.coingecko.com/api/v3';

function cgBase(env) { return env.COINGECKO_KEY ? CG_PRO : CG_FREE; }
function cgHeaders(env) { return env.COINGECKO_KEY ? { 'x-cg-pro-api-key': env.COINGECKO_KEY } : {}; }

export async function priceTokens(chain, env, contracts, fiat) {
  if (!contracts.length) return {};
  const fiatLow = fiat.toLowerCase();
  const cacheKey = `px:${chain.id}:${fiatLow}:${contracts.sort().join(',')}`;
  const cached = await cacheGet(env, cacheKey);
  if (cached) return cached;

  const url = `${cgBase(env)}/simple/token_price/${chain.coingecko}?contract_addresses=${contracts.join(',')}&vs_currencies=${fiatLow}`;
  const r = await fetch(url, { headers: cgHeaders(env) });
  if (!r.ok) return {};
  const j = await r.json();
  await cacheSet(env, cacheKey, j, 60);
  return j;
}

import { getNativeBalance, getErc20Balances } from '../lib/providers.js';
import { priceTokens, priceNative } from '../lib/prices.js';

const isAddress = (a) => /^0x[a-fA-F0-9]{40}$/.test(a || '');

async function scanChain(chain, env, address, fiat) {
  const [native, tokens] = await Promise.all([
    getNativeBalance(chain, env, address).catch(() => 0),
    getErc20Balances(chain, env, address).catch(() => []),
  ]);

  const [nativePx, tokenPxs] = await Promise.all([
    priceNative(chain, env, fiat).catch(() => 0),
    priceTokens(chain, env, tokens.map((t) => t.contract), fiat).catch(() => ({})),
  ]);

  const tokenRows = tokens.map((t) => {
    const px = tokenPxs[t.contract.toLowerCase()]?.[fiat.toLowerCase()] ?? 0;
    return { ...t, chain: chain.id, priceFiat: px, valueFiat: px * t.amount };
  });

  const nativeRow = native > 0 ? [{
    contract: 'native',
    symbol: chain.nativeSymbol,
    name: `${chain.name} native`,
    chain: chain.id,
    decimals: 18,
    amount: native,
    priceFiat: nativePx,
    valueFiat: nativePx * native,
    logo: null,
  }] : [];

  const allTokens = [...nativeRow, ...tokenRows];
  return {
    chain: chain.id,
    chainName: chain.name,
    chainId: chain.chainId,
    tier: chain.tier,
    tokens: allTokens.sort((a, b) => b.valueFiat - a.valueFiat),
    totalFiat: allTokens.reduce((s, t) => s + (t.valueFiat || 0), 0),
  };
}

export async function handlePortfolio(request, env) {
  const url = new URL(request.url);
  const address = (url.searchParams.get('address') || '').toLowerCase();
  const fiat = (url.searchParams.get('fiat') || 'USD').toUpperCase();
  const chainFilter = url.searchParams.get('chains'); // optional CSV

  if (!isAddress(address)) {
    return json({ error: 'invalid address' }, 400);
  }

  const chainsToScan = chainFilter
    ? CHAINS.filter((c) => chainFilter.split(',').includes(c.id))
    : CHAINS;

  const perChain = await Promise.all(chainsToScan.map((c) =>
    scanChain(c, env, address, fiat).catch((err) => ({
      chain: c.id, chainName: c.name, chainId: c.chainId, tier: c.tier,
      tokens: [], totalFiat: 0, error: String(err),
    }))
  ));

  const portfolioFiat = perChain.reduce((s, c) => s + c.totalFiat, 0);
  const activeChains = perChain.filter((c) => c.totalFiat > 0).length;
  const totalTokens = perChain.reduce((s, c) => s + c.tokens.length, 0);

  return json({
    address,
    fiat,
    portfolioFiat,
    activeChains,
    totalTokens,
    chains: perChain,
    updatedAt: new Date().toISOString(),
  });
}

function json(data, status = 200) {
  return new Response(JSON.stringify(data, null, 2), {
    status,
    headers: {
      'content-type': 'application/json',
      'cache-control': 'public, max-age=30',
      'access-control-allow-origin': '*',
    },
  });
}

// getUserAccountData(address) returns:
//   totalCollateralBase, totalDebtBase, availableBorrowsBase,
//   currentLiquidationThreshold, ltv, healthFactor
const GET_USER_ACCOUNT_DATA = '0xbf92857c'; // function selector

async function readAaveV3(chain, env, address) {
  const pool = AAVE_V3_POOL[chain.id];
  if (!pool || !chain.alchemy) return null;
  const url = `https://${chain.alchemy}.g.alchemy.com/v2/${env.ALCHEMY_KEY}`;
  const data = GET_USER_ACCOUNT_DATA + address.replace(/^0x/, '').padStart(64, '0');
  const r = await fetch(url, {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({
      jsonrpc: '2.0', id: 1, method: 'eth_call',
      params: [{ to: pool, data }, 'latest'],
    }),
  });
  const j = await r.json();
  if (!j.result || j.result === '0x') return null;
  const hex = j.result.replace(/^0x/, '');
  if (hex.length < 64 * 6) return null;
  const parse = (i) => BigInt('0x' + hex.slice(i * 64, (i + 1) * 64));
  const totalCollateral = Number(parse(0)) / 1e8; // USD with 8 decimals
  const totalDebt = Number(parse(1)) / 1e8;
  const ltv = Number(parse(3)) / 10000;
  const healthFactor = Number(parse(5)) / 1e18;
  if (totalCollateral === 0 && totalDebt === 0) return null;
  return {
    protocol: 'Aave v3',
    chain: chain.id,
    type: 'lending',
    suppliedUsd: totalCollateral,
    borrowedUsd: totalDebt,
    netUsd: totalCollateral - totalDebt,
    ltv,
    healthFactor: healthFactor > 1e18 ? null : healthFactor,
    liquidationRisk: healthFactor < 1.5 ? 'high' : healthFactor < 2 ? 'medium' : 'low',
  };
}

// DefiLlama's /api/v1/positions/{address} returns yield positions tracked by them
async function readDefiLlamaPositions(env, address) {
  const url = `https://yields.llama.fi/poolsByAddress/${address}`;
  const r = await fetch(url);
  if (!r.ok) return [];
  const j = await r.json();
  return (j.data || []).map((p) => ({
    protocol: p.project,
    chain: p.chain?.toLowerCase(),
    type: 'yield',
    poolName: p.symbol,
    apy: p.apy,
    suppliedUsd: p.tvlUsd || 0,
    borrowedUsd: 0,
    netUsd: p.tvlUsd || 0,
  }));
}

export async function handleDefiPositions(request, env) {
  const url = new URL(request.url);
  const address = (url.searchParams.get('address') || '').toLowerCase();
  if (!isAddress(address)) {
    return new Response(JSON.stringify({ error: 'invalid address' }), {
      status: 400, headers: { 'content-type': 'application/json' },
    });
  }

  const aaveResults = await Promise.all(
    CHAINS.filter((c) => AAVE_V3_POOL[c.id])
          .map((c) => readAaveV3(c, env, address).catch(() => null))
  );
  const llamaResults = await readDefiLlamaPositions(env, address).catch(() => []);

  const positions = [...aaveResults.filter(Boolean), ...llamaResults];
  const totalSupplied = positions.reduce((s, p) => s + (p.suppliedUsd || 0), 0);
  const totalBorrowed = positions.reduce((s, p) => s + (p.borrowedUsd || 0), 0);
  const atRisk = positions.filter((p) => p.liquidationRisk === 'high').length;

  return new Response(JSON.stringify({
    address,
    positions,
    totalSupplied,
    totalBorrowed,
    netDefiValue: totalSupplied - totalBorrowed,
    positionCount: positions.length,
    atRiskCount: atRisk,
    updatedAt: new Date().toISOString(),
  }, null, 2), {
    headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },
  });
}

async function getOpenSeaFloor(contract, env) {
  if (!env.OPENSEA_KEY) return null;
  const url = `https://api.opensea.io/api/v2/collections/${contract}/stats`;
  const r = await fetch(url, { headers: { 'X-API-KEY': env.OPENSEA_KEY } });
  if (!r.ok) return null;
  const j = await r.json();
  return j.total?.floor_price || null;
}

export async function handleNfts(request, env) {
  const url = new URL(request.url);
  const address = (url.searchParams.get('address') || '').toLowerCase();
  const fiat = (url.searchParams.get('fiat') || 'USD').toUpperCase();
  if (!isAddress(address)) {
    return new Response(JSON.stringify({ error: 'invalid address' }), {
      status: 400, headers: { 'content-type': 'application/json' },
    });
  }

  const perChainNfts = await Promise.all(
    CHAINS.map((c) => getNfts(c, env, address).catch(() => []))
  );
  const allNfts = perChainNfts.flat();

  // Group by collection so we only look up floor once
  const byCollection = new Map();
  for (const n of allNfts) {
    const k = `${n.chain}:${n.contract}`;
    if (!byCollection.has(k)) byCollection.set(k, { ...n, items: [] });
    byCollection.get(k).items.push({ tokenId: n.tokenId, title: n.title, image: n.image });
  }

  const ethPrice = await priceNative(CHAINS[0], env, fiat).catch(() => 0);

  const collections = await Promise.all([...byCollection.values()].map(async (c) => {
    const floorEth = c.floorEth ?? null;
    const floorFiat = floorEth ? floorEth * ethPrice : null;
    return {
      chain: c.chain,
      contract: c.contract,
      collection: c.collection,
      itemCount: c.items.length,
      floorEth,
      floorFiat,
      estimatedValueFiat: floorFiat ? floorFiat * c.items.length : null,
      items: c.items.slice(0, 10),
    };
  }));

  const totalEstimatedValue = collections.reduce((s, c) => s + (c.estimatedValueFiat || 0), 0);

  return new Response(JSON.stringify({
    address,
    fiat,
    totalNfts: allNfts.length,
    totalCollections: collections.length,
    totalEstimatedValueFiat: totalEstimatedValue,
    collections: collections.sort((a, b) => (b.estimatedValueFiat || 0) - (a.estimatedValueFiat || 0)),
    updatedAt: new Date().toISOString(),
  }, null, 2), {
    headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },
  });
}

// worker/scoring/factors.js
// Each factor returns 0..maxWeight. Sum is mapped to FICO-style band.

export function onChainActivityFactor(portfolio) {
  const tokenCount = portfolio.totalTokens || 0;
  return Math.min(20, tokenCount * 1.2);
}

export function multiChainDiversityFactor(portfolio) {
  const active = portfolio.activeChains || 0;
  return Math.min(15, active * 3);
}

export function nativeLiquidityFactor(portfolio) {
  const v = portfolio.portfolioFiat || 0;
  if (v >= 10000) return 20;
  if (v >= 1000) return 15;
  if (v >= 100) return 8;
  if (v >= 10) return 3;
  return 1;
}

export function walletAgeFactor(firstTxTimestamp) {
  if (!firstTxTimestamp) return 5;
  const days = (Date.now() - firstTxTimestamp) / (1000 * 60 * 60 * 24);
  if (days >= 1095) return 15; // 3+ years
  if (days >= 365) return 12;
  if (days >= 180) return 8;
  if (days >= 30) return 4;
  return 1;
}

export function contractDiversityFactor(portfolio) {
  const contracts = new Set();
  for (const c of portfolio.chains || []) {
    for (const t of c.tokens || []) {
      if (t.contract !== 'native') contracts.add(t.contract);
    }
  }
  return Math.min(15, contracts.size * 0.8);
}

export function tokenDiversityFactor(portfolio) {
  const symbols = new Set();
  for (const c of portfolio.chains || []) {
    for (const t of c.tokens || []) symbols.add(t.symbol);
  }
  return Math.min(10, symbols.size * 0.7);
}

export function defiEngagementFactor(positions) {
  const count = positions.positionCount || 0;
  const atRisk = positions.atRiskCount || 0;
  const base = Math.min(10, count * 2);
  return Math.max(0, base - atRisk * 3);
}

export function nftHoldingsFactor(nftData) {
  const v = nftData.totalEstimatedValueFiat || 0;
  if (v >= 50000) return 5;
  if (v >= 5000) return 4;
  if (v >= 500) return 3;
  if (v >= 50) return 2;
  if ((nftData.totalNfts || 0) > 0) return 1;
  return 0;
}

export function toFicoScore(raw) {
  const clamped = Math.max(0, Math.min(100, raw));
  return Math.round(300 + (clamped / 100) * 550);
}

export function bandFor(score) {
  if (score >= 800) return { band: 'Excellent', color: '#10b981' };
  if (score >= 740) return { band: 'Very Good', color: '#22c55e' };
  if (score >= 670) return { band: 'Good', color: '#84cc16' };
  if (score >= 580) return { band: 'Fair', color: '#f59e0b' };
  return { band: 'Poor', color: '#ef4444' };

  const fiat = (url.searchParams.get('fiat') || 'USD').toUpperCase();

// worker/handlers/alerts.js
// Generates real-time alerts. For now, computes from current state;
// in Task 14 we wire this to a Durable Object for historical tracking.

import { handlePortfolio } from './portfolio.js';
import { handleDefiPositions } from './defi-positions.js';

const isAddress = (a) => /^0x[a-fA-F0-9]{40}$/.test(a || '');

const DUST_THRESHOLD_USD = 0.01;

// Hardcoded suspicious-token denylist; in production, sync from a public list.
const SUSPICIOUS_PATTERNS = [
  /visit.*claim/i,
  /\.com$/i,
  /\.io$/i,
  /airdrop/i,
  /reward/i,
];

export async function handleAlerts(request, env) {
  const url = new URL(request.url);
  const address = (url.searchParams.get('address') || '').toLowerCase();
  if (!isAddress(address)) {
    return new Response(JSON.stringify({ error: 'invalid address' }), {
      status: 400, headers: { 'content-type': 'application/json' },
    });
  }

  const baseUrl = url.origin;
  const [portRes, posRes] = await Promise.all([
    handlePortfolio(new Request(`${baseUrl}/api/portfolio?address=${address}&fiat=USD`), env),
    handleDefiPositions(new Request(`${baseUrl}/api/defi-positions?address=${address}`), env),
  ]);
  const [portfolio, positions] = await Promise.all([portRes.json(), posRes.json()]);

  const alerts = [];

  // 1. Health-factor alerts
  for (const p of (positions.positions || [])) {
    if (p.healthFactor != null && p.healthFactor < 1.2) {
      alerts.push({
        severity: 'critical',
        type: 'liquidation_risk',
        title: `${p.protocol} position near liquidation`,
        message: `Health factor: ${p.healthFactor.toFixed(2)} on ${p.chain}. Add collateral or repay.`,
        actionUrl: `https://app.aave.com`,
      });
    } else if (p.healthFactor != null && p.healthFactor < 1.5) {
      alerts.push({
        severity: 'warning',
        type: 'low_health',
        title: `${p.protocol} health factor low`,
        message: `Health factor: ${p.healthFactor.toFixed(2)} on ${p.chain}.`,
      });
    }
  }

  // 2. Suspicious tokens (likely scam airdrops)
  for (const c of (portfolio.chains || [])) {
    for (const t of (c.tokens || [])) {
      const looksScam = SUSPICIOUS_PATTERNS.some((re) => re.test(t.name || '') || re.test(t.symbol || ''));
      if (looksScam) {
        alerts.push({
          severity: 'warning',
          type: 'suspicious_token',
          title: `Suspicious token detected: ${t.symbol}`,
          message: `On ${c.chainName}. Do NOT interact — likely scam airdrop. Hide or revoke approvals.`,
          tokenContract: t.contract,
          chain: c.chain,
        });
      }
    }
  }

  // 3. Dust (informational)
  let dustCount = 0;
  for (const c of (portfolio.chains || [])) {
    for (const t of (c.tokens || [])) {
      if (t.valueFiat > 0 && t.valueFiat < DUST_THRESHOLD_USD) dustCount++;
    }
  }
  if (dustCount >= 5) {
    alerts.push({
      severity: 'info',
      type: 'dust',
      title: `${dustCount} dust tokens detected`,
      message: 'Consider sweeping to consolidate.',
    });
  }

  // 4. Single-chain concentration
  const sorted = (portfolio.chains || []).slice().sort((a, b) => b.totalFiat - a.totalFiat);
  if (sorted[0]?.totalFiat > 0 && sorted[0].totalFiat / portfolio.portfolioFiat > 0.9) {
    alerts.push({
      severity: 'info',
      type: 'concentration',
      title: 'Portfolio concentrated on one chain',
      message: `${(sorted[0].totalFiat / portfolio.portfolioFiat * 100).toFixed(0)}% on ${sorted[0].chainName}.`,
    });
  }

  return new Response(JSON.stringify({
    address,
    alertCount: alerts.length,
    criticalCount: alerts.filter((a) => a.severity === 'critical').length,
    alerts,
    updatedAt: new Date().toISOString(),
  }, null, 2), {
    headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },
  });
}

export async function handleProtocols(request, env) {
  const url = new URL(request.url);
  const category = url.searchParams.get('category');
  const chain = url.searchParams.get('chain');
  const minScore = Number(url.searchParams.get('minScore') || 0);

  // Enrich with TVL from DefiLlama (cached in real deploy)
  let tvls = {};
  try {
    const r = await fetch('https://api.llama.fi/protocols');
    if (r.ok) {
      const list = await r.json();
      tvls = Object.fromEntries(list.map((p) => [p.slug, p.tvl]));
    }
  } catch {}

  let results = CATALOG.map((p) => ({ ...p, tvlUsd: tvls[p.llamaSlug] || null }));
  if (category) results = results.filter((p) => p.category === category);
  if (chain) results = results.filter((p) => p.chains.includes(chain));
  if (minScore) results = results.filter((p) => p.safetyScore >= minScore);

  return new Response(JSON.stringify({
    count: results.length,
    protocols: results.sort((a, b) => b.safetyScore - a.safetyScore),
    categories: [...new Set(CATALOG.map((p) => p.category))],
    updatedAt: new Date().toISOString(),
  }, null, 2), {
    headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },
  });
}

export default {
  async fetch(request, env) {
    const url = new URL(request.url);

    if (request.method === 'OPTIONS') {
      return new Response(null, {
        headers: {
          'access-control-allow-origin': '*',
          'access-control-allow-methods': 'GET,POST,OPTIONS',
          'access-control-allow-headers': 'content-type, authorization',
          'access-control-max-age': '86400',
        },
      });
    }

    const handler = ROUTES[url.pathname];
    if (handler) {
      try {
        return await handler(request, env);
      } catch (err) {
        return new Response(JSON.stringify({ error: String(err.message || err) }), {
          status: 500,
          headers: { 'content-type': 'application/json', 'access-control-allow-origin': '*' },
        });
      }
    }

    return new Response('Not Found', { status: 404 });
  },
};

Got it — that changes the architecture significantly. You're describing a fully static, edge-computed stack with no traditional backend server. Let me reframe everything around that constraint, because it's actually a clever choice for this product (cheap, fast, censorship-resistant, fits the no-KYC ethos) but it has hard limits you need to design around.
The stack, clarified
Replit = the build environment / IDE / AI agent that writes the code and pushes to GitHub.
GitHub = source of truth, CI/CD via Actions, issue tracking.
GitHub Pages (Jekyll) = static site hosting for the marketing site, methodology docs, public dashboards, backtest pages.
Cloudflare Workers / Pages = the actual compute layer. This is where scoring, API endpoints, caching, auth, paywall logic, and webhook delivery live.
Cloudflare Web3 Gateways = Ethereum + IPFS gateways, so you don't pay Alchemy/Infura for RPC. Also Cloudflare's distributed web hosting via IPFS if you want a .eth domain mirror.
Cloudflare D1 (SQLite at edge) + KV (cache) + R2 (object storage for PDFs/exports) + Durable Objects (for stateful things like rate limits and webhook queues) + Queues (for async jobs).
What this stack can't do — design constraints to accept
No long-running compute. Workers cap at 30s CPU on paid plan, 50ms free. ZK proof generation, ML training, and heavy backtests cannot run here. Solution: precompute offline (GitHub Actions cron), commit results as JSON to the repo, serve as static files.
No Solana RPC via Cloudflare Web3. Cloudflare's Web3 gateway is Ethereum-only. Solana/Cosmos need Helius/Triton/QuickNode. Budget for this — it's the one paid dependency you can't eliminate.
No persistent in-memory state. Workers are stateless. Use Durable Objects or D1 for anything that needs to persist across requests.
Jekyll is build-time only. Anything dynamic (live scores, user dashboards) must be client-side JS hitting Workers, not Jekyll templates.
No traditional sessions. Auth must be SIWE (Sign-In With Ethereum) → JWT in cookie, verified by Worker on each request.

Paywall strategy (the part you asked about explicitly)
Since you have no KYC and no Stripe-friendly user accounts in the traditional sense, the paywall has to be Web3-native. Three tiers, three mechanisms:
Free tier — public, rate-limited
10 wallet scores per IP per day, hashed IP in KV.
Single-chain only (Ethereum mainnet).
No API key needed.
Cached scores, 1-hour staleness OK.
Watermarked PDF exports.
Pro tier — $19/mo or 0.01 ETH/mo, paid on-chain
Unlimited wallet scores, all supported chains.
API key with 1000 req/day.
Real-time alerts via Telegram/Discord/email.
Synced watchlist across devices.
Clean PDF exports.
Mechanism: user pays a Coinbase Commerce charge, Stripe checkout, or sends payment to a subscription smart contract (Unlock Protocol or Hypersub). Worker verifies on-chain payment via Cloudflare Eth Gateway → issues signed JWT good for 30 days.
Institutional / API tier — $499+/mo, custom
Higher rate limits (100k+ req/day).
Webhook delivery with HMAC signing.
Bulk wallet scoring endpoint.
White-label PDF reports.
SLA + priority queue.
Mechanism: invoiced via Stripe or paid in stables to multisig; API key issued manually.
Recommendation: use Unlock Protocol or Hypersub as the primary on-chain paywall. They're battle-tested, give you NFT-based subscription keys (so users can resell/transfer), and verification is one eth_call in a Worker. Stripe as the fiat fallback for institutions who can't pay in crypto.
The Worker auth flow looks like:
1. User signs SIWE message → Worker issues JWT (free tier).
2. User upgrades → pays Unlock Protocol → receives NFT key.
3. Worker checks key ownership via getHasValidKey() on every Pro endpoint hit, caches result in KV for 5min.
4. JWT carries tier claim; Worker re-verifies key on cache miss.

Big things to fix / get right early
Don't put secrets in the repo. Cloudflare Worker secrets via wrangler secret put, never in wrangler.toml. Anti-sybil heuristics in a separate private repo, deployed as a Worker that the public Worker calls internally.
JWT secret rotation — bake a kid into JWTs from day one.
D1 has size limits (10GB per DB, currently). Plan for partitioning by chain or by user-cohort once you grow.
Cloudflare Eth Gateway has rate limits too — not infinite. Cache aggressively in KV; never hit it twice for the same (block, address) tuple.
Jekyll on Pages doesn't support every plugin — use jekyll-redirect-from, jekyll-sitemap, jekyll-feedand forget custom plugins unless you switch to GitHub Actions-built Pages.
CORS. Workers must explicitly set CORS headers; the dashboard at defiscoring.com calling api.defiscoring.com will fail silently otherwise.
Score versioning in every response — methodology_version, computed_at_block, chain_coverage array. Lenders need this for audit trails.
Webhook signing day one. HMAC with rotating secret per subscriber.
Idempotency keys on POST endpoints so retries don't double-charge or double-trigger.
Static fallback for the dashboard. If Workers go down, the Jekyll site should still load and explain the outage. Don't have a hard dependency from marketing pages to live API.
One pushback on this stack choice
The biggest risk: Cloudflare Workers' 30s CPU limit will hurt you when scoring whale wallets with 50k+ transactions. You'll need to chunk indexing into Queue jobs and return "score pending" responses, then poll. Design the API for async-by-default from day one — don't promise synchronous scoring even if it works for 95% of wallets today.
