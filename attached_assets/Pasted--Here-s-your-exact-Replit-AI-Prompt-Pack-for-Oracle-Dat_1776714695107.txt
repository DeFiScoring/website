```
Create a new file exactly at assets/js/oracle-integrity.js with this complete, ready-to-use code:

// assets/js/oracle-integrity.js
// Oracle & Data Integrity - Module 5 for RWA Scoring

const ORACLE_DATA = {
  'BlackRock/Securitize': {
    oracleProvider: 'Chainlink + RedStone',
    riskScoreBase: 810,
    latencyMs: 450,
    redemptionRisk: 'Low',
    failurePatterns: [2, 1, 0, 3, 1, 0, 2], // past 7 days
    freshnessHours: [0.2, 0.3, 0.4, 0.25, 0.35],
    navAccuracy: 99.8
  },
  'Ondo Finance': {
    oracleProvider: 'Chainlink',
    riskScoreBase: 760,
    latencyMs: 680,
    redemptionRisk: 'Medium',
    failurePatterns: [4, 2, 5, 1, 3, 4, 2],
    freshnessHours: [0.8, 1.1, 0.9, 0.7, 1.0],
    navAccuracy: 98.5
  },
  'Centrifuge': {
    oracleProvider: 'Custom + Pyth',
    riskScoreBase: 590,
    latencyMs: 2400,
    redemptionRisk: 'High',
    failurePatterns: [8, 12, 5, 9, 15, 7, 11],
    freshnessHours: [4.5, 6.2, 3.8, 5.1, 4.9],
    navAccuracy: 94.2
  }
};

async function calculateOracleScore(issuerName) {
  const data = ORACLE_DATA[issuerName] || { riskScoreBase: 550, oracleProvider: 'Unknown', latencyMs: 2000, redemptionRisk: 'High', failurePatterns: [5,5,5], freshnessHours: [3,3,3], navAccuracy: 95 };

  let score = data.riskScoreBase;
  // Penalty for high latency and failures
  if (data.latencyMs > 1000) score -= 80;
  if (data.redemptionRisk === 'High') score -= 70;
  score -= Math.floor(data.failurePatterns.reduce((a,b)=>a+b,0) * 8);

  score = Math.min(850, Math.max(300, Math.floor(score)));

  const grade = score >= 800 ? 'A+' : score >= 700 ? 'A' : score >= 600 ? 'B' : score >= 500 ? 'C' : 'D';
  const tier = score >= 750 ? 'Low Risk' : score >= 600 ? 'Moderate Risk' : 'High Risk';

  return {
    score,
    grade,
    tier,
    oracleProvider: data.oracleProvider,
    latencyMs: data.latencyMs,
    redemptionRisk: data.redemptionRisk,
    failurePatterns: data.failurePatterns,
    freshnessHours: data.freshnessHours,
    navAccuracy: data.navAccuracy
  };
}

async function renderOracleIntegrity() {
  const container = document.getElementById('oracle-integrity-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to assess oracle & data integrity for your RWA holdings</div>`;
    return;
  }

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Analyzing pricing oracles, NAV feeds, and redemption reliability...</p></div>`;

  try {
    const issuerName = "BlackRock/Securitize"; // In production, detect dynamically from wallet
    const result = await calculateOracleScore(issuerName);

    const html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-8">
        <!-- Oracle Risk Score Card -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8 text-center">
          <h2 class="text-sm uppercase tracking-widest text-indigo-600 mb-2">Oracle & Data Integrity Score</h2>
          <div class="text-8xl font-bold text-indigo-600">${result.score}</div>
          <div class="text-4xl font-semibold mt-1 mb-6">${result.grade}</div>
          <div class="inline-block px-6 py-2 bg-indigo-100 dark:bg-indigo-900 text-indigo-700 dark:text-indigo-200 rounded-3xl text-lg">${result.tier}</div>
          <p class="mt-8 text-sm text-gray-500">Oracle: ${result.oracleProvider}</p>
        </div>

        <!-- Latency & Redemption Risk -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4 flex items-center gap-2">⚡ Latency & Redemption Risk</h3>
          <div class="space-y-6">
            <div>
              <div class="flex justify-between text-sm mb-1"><span>Oracle Latency</span><span class="font-medium">${result.latencyMs} ms</span></div>
              <div class="h-2 bg-gray-200 dark:bg-gray-700 rounded-full overflow-hidden"><div class="h-2 ${result.latencyMs < 1000 ? 'bg-green-500' : 'bg-orange-500'}" style="width: ${Math.min(100, result.latencyMs/30)}%"></div></div>
            </div>
            <div class="flex justify-between items-center">
              <span>Redemption Risk</span>
              <span class="px-5 py-1 rounded-3xl text-sm font-medium ${result.redemptionRisk === 'Low' ? 'bg-green-100 text-green-700' : result.redemptionRisk === 'Medium' ? 'bg-yellow-100 text-yellow-700' : 'bg-red-100 text-red-700'}">${result.redemptionRisk}</span>
            </div>
            <div class="text-xs text-gray-500">NAV Accuracy: <strong>${result.navAccuracy}%</strong></div>
          </div>
        </div>

        <!-- Oracle Failure Patterns -->
        <div class="lg:col-span-1 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Oracle Failure Patterns (Last 7 days)</h3>
          <canvas id="oracleFailureChart" height="240"></canvas>
        </div>

        <!-- Data Freshness Stats -->
        <div class="lg:col-span-3 bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-8">
          <h3 class="font-semibold mb-4">Data Freshness Stats (NAV feed update frequency)</h3>
          <canvas id="freshnessChart" height="180"></canvas>
        </div>
      </div>
    `;

    container.innerHTML = html;

    setTimeout(() => {
      // Failure Patterns Bar Chart
      new Chart(document.getElementById('oracleFailureChart'), {
        type: 'bar',
        data: {
          labels: ['Day 1', 'Day 2', 'Day 3', 'Day 4', 'Day 5', 'Day 6', 'Day 7'],
          datasets: [{
            label: 'Failures',
            data: result.failurePatterns,
            backgroundColor: '#6366f1',
            borderRadius: 6
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } }, scales: { y: { beginAtZero: true } } }
      });

      // Freshness Line Chart
      new Chart(document.getElementById('freshnessChart'), {
        type: 'line',
        data: {
          labels: ['12h', '24h', '36h', '48h', '60h'],
          datasets: [{
            label: 'Hours since last update',
            data: result.freshnessHours,
            borderColor: '#6366f1',
            tension: 0.4,
            borderWidth: 3,
            pointRadius: 5
          }]
        },
        options: { responsive: true, plugins: { legend: { display: false } } }
      });
    }, 300);

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Could not fetch oracle data. Please try again.</p>`;
  }
}

window.renderOracleIntegrity = renderOracleIntegrity;
```
