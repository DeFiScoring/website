### Prompt 1: Create the data-aggregation.js file
```
Create a new file exactly at assets/js/data-aggregation.js with this complete, ready-to-use code for real-time DeFi data aggregation:

// assets/js/data-aggregation.js
// Real-time Data Aggregation: DeFiLlama + CoinGecko + The Graph

async function fetchDeFiData() {
  const container = document.getElementById('data-aggregation-container');
  if (!container) return;

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Fetching live market data...</p></div>`;

  try {
    // 1. DeFiLlama - Top protocols by TVL
    const llamaRes = await fetch('https://api.llama.fi/protocols');
    const llamaData = await llamaRes.json();
    const topProtocols = llamaData.slice(0, 6).map(p => ({
      name: p.name,
      tvl: (p.tvl / 1e9).toFixed(2) + 'B',
      category: p.category || 'DeFi'
    }));

    // 2. CoinGecko - Live prices
    const cgRes = await fetch('https://api.coingecko.com/api/v3/simple/price?ids=bitcoin,ethereum,usd-coin&vs_currencies=usd');
    const cgData = await cgRes.json();

    // 3. The Graph - Aave V3 Ethereum sample data (recent borrows + total supply)
    const graphQuery = {
      query: `{
        markets(first: 5, orderBy: totalSupply, orderDirection: desc) {
          name
          totalSupply
          totalBorrows
        }
      }`
    };
    const graphRes = await fetch('https://api.thegraph.com/subgraphs/name/aave/protocol-v3', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(graphQuery)
    });
    const graphData = await graphRes.json();

    // Render everything
    let html = `
      <div class="grid grid-cols-1 lg:grid-cols-3 gap-6">
        <!-- DeFiLlama TVL -->
        <div class="bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-6">
          <h3 class="font-semibold mb-4 flex items-center gap-2">📊 Top TVL Protocols (DeFiLlama)</h3>
          <ul class="space-y-3">
            ${topProtocols.map(p => `
              <li class="flex justify-between text-sm">
                <span>${p.name}</span>
                <span class="font-medium">${p.tvl}</span>
              </li>
            `).join('')}
          </ul>
        </div>

        <!-- CoinGecko Prices -->
        <div class="bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-6">
          <h3 class="font-semibold mb-4 flex items-center gap-2">💰 Live Prices (CoinGecko)</h3>
          <div class="space-y-4">
            <div class="flex justify-between"><span>BTC</span><span class="font-mono">$${cgData.bitcoin.usd.toLocaleString()}</span></div>
            <div class="flex justify-between"><span>ETH</span><span class="font-mono">$${cgData.ethereum.usd.toLocaleString()}</span></div>
            <div class="flex justify-between"><span>USDC</span><span class="font-mono">$${cgData['usd-coin'].usd}</span></div>
          </div>
        </div>

        <!-- The Graph - Aave V3 Insights -->
        <div class="bg-white dark:bg-gray-900 rounded-3xl shadow-xl p-6">
          <h3 class="font-semibold mb-4 flex items-center gap-2">🔗 Aave V3 (The Graph)</h3>
          <ul class="space-y-3 text-sm">
            ${graphData.data.markets.map(m => `
              <li>
                <span class="font-medium">${m.name}</span><br>
                <span class="text-green-600">Supply: $${(m.totalSupply / 1e18).toFixed(0)}B</span><br>
                <span class="text-orange-600">Borrows: $${(m.totalBorrows / 1e18).toFixed(0)}B</span>
              </li>
            `).join('')}
          </ul>
        </div>
      </div>
    `;

    container.innerHTML = html;

  } catch (error) {
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Data fetch failed. Please check console.</p>`;
    console.error('Data aggregation error:', error);
  }
}

// Auto-refresh every 60 seconds (optional)
window.fetchDeFiData = fetchDeFiData;
setInterval(() => { if (document.getElementById('data-aggregation-container')) fetchDeFiData(); }, 60000);
```
