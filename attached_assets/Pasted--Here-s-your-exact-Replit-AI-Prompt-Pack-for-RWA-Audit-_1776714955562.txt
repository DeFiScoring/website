### Prompt 1: Create the RWA Audit & Export Toolkit JavaScript file
```
Create a new file exactly at assets/js/rwa-audit-toolkit.js with this complete, ready-to-use code:

// assets/js/rwa-audit-toolkit.js
// RWA Audit & Export Toolkit - Module 9 (Revenue-generating feature)

async function generateAuditReport() {
  const container = document.getElementById('audit-toolkit-container');
  if (!container) return;

  container.innerHTML = `<div class="text-center py-8"><p class="text-gray-500">Generating institutional-grade RWA audit report...</p></div>`;

  try {
    // Aggregate data from all previous modules (demo values)
    const auditData = {
      timestamp: new Date().toISOString(),
      wallet: window.userWallet || "0xDemoWallet",
      totalRwaExposure: 243000,
      portfolioRiskScore: 785,
      summary: {
        assetRisk: 820,
        issuerDiligence: 810,
        legalCompliance: 820,
        custodyPor: 830,
        oracleIntegrity: 810,
        liquidityRedemption: 820,
        yieldRiskAdjusted: 810
      }
    };

    // Generate PDF with jsPDF
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF();
    
    doc.setFontSize(22);
    doc.text("RWA Institutional Audit Report", 20, 30);
    doc.setFontSize(12);
    doc.text(`Generated: ${new Date().toLocaleDateString()}`, 20, 45);
    doc.text(`Wallet: ${auditData.wallet.substring(0,8)}...`, 20, 55);

    doc.setFontSize(14);
    doc.text("Portfolio Summary", 20, 75);
    doc.setFontSize(11);
    doc.text(`Total RWA Exposure: $${(auditData.totalRwaExposure / 1000).toFixed(0)}k`, 20, 85);
    doc.text(`Overall Portfolio Risk Score: ${auditData.portfolioRiskScore}`, 20, 95);

    // Module scores
    let y = 115;
    Object.entries(auditData.summary).forEach(([key, score]) => {
      doc.text(`${key.replace(/([A-Z])/g, ' $1').toUpperCase()}: ${score}`, 20, y);
      y += 10;
    });

    doc.save(`RWA-Audit-Report-${new Date().toISOString().slice(0,10)}.pdf`);

    // Also generate CSV for raw dataset export
    const csvContent = `Asset,Issuer,RiskScore\nBUIDL,BlackRock/Securitize,820\nOUSG,Ondo Finance,780\nCentrifuge Pool,Centrifuge,580`;
    const blob = new Blob([csvContent], { type: 'text/csv' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `RWA-Raw-Dataset-${new Date().toISOString().slice(0,10)}.csv`;
    a.click();

    // Log anonymized audit event to your Market Intelligence
    if (window.logMarketEvent) {
      window.logMarketEvent('rwa_audit_generated', auditData.portfolioRiskScore, 'Low Risk', { modulesCompleted: 9 });
    }

    container.innerHTML = `
      <div class="bg-green-100 dark:bg-green-900 p-8 rounded-3xl text-center">
        <h3 class="text-2xl font-bold mb-4">✅ Institutional Audit Report Generated!</h3>
        <p class="mb-6">PDF + Raw CSV dataset downloaded.</p>
        <p class="text-sm text-gray-600 dark:text-gray-400">This feature is part of your premium institutional offering.</p>
        <button onclick="generateAuditReport()" class="mt-6 bg-purple-600 hover:bg-purple-700 text-white px-8 py-4 rounded-2xl">Regenerate Full Report</button>
      </div>
    `;

  } catch (e) {
    console.error(e);
    container.innerHTML = `<p class="text-red-500 text-center py-8">⚠️ Error generating audit report. Please try again.</p>`;
  }
}

async function renderRWAAuditToolkit() {
  const container = document.getElementById('audit-toolkit-container');
  if (!container) return;

  if (!window.userWallet) {
    container.innerHTML = `<div class="text-center py-12 text-gray-500">Connect your wallet to generate institutional RWA audit reports and export raw datasets</div>`;
    return;
  }

  container.innerHTML = `
    <div class="text-center py-12">
      <h3 class="text-2xl font-semibold mb-4">RWA Audit & Export Toolkit</h3>
      <p class="max-w-md mx-auto mb-8 text-gray-600 dark:text-gray-400">One-click professional audit reports + CSV exports for institutions. Full anonymized logs saved for your R&amp;D and data resale.</p>
      <button onclick="generateAuditReport()" 
              class="bg-gradient-to-r from-purple-600 to-violet-600 text-white font-bold py-5 px-12 rounded-3xl text-xl shadow-xl hover:shadow-2xl transition-all">
        Generate Full Institutional Audit Report + CSV
      </button>
      <p class="text-xs text-gray-500 mt-6">💼 This is your direct revenue stream — premium institutional feature</p>
    </div>
  `;
}

// Make global
window.renderRWAAuditToolkit = renderRWAAuditToolkit;
window.generateAuditReport = generateAuditReport;
```
