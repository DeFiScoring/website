### Prompt 2: Replace the Worker code (full multi-turn AI chatbot + PDF + email)
```
Replace the entire content of src/index.js with this complete Cloudflare AI Worker for the gated Risk Profile Chatbot:

export default {
  async fetch(request, env) {
    if (request.method !== "POST") return new Response("Method not allowed", { status: 405 });

    const { action, sessionId, message, email, consent } = await request.json();

    // Consent & lead capture
    if (action === "submit-consent" && email) {
      await env.D1.prepare(`INSERT INTO leads (email, consented_at, source) VALUES (?, ?, 'chatbot') ON CONFLICT(email) DO UPDATE SET consented_at = ?`)
        .bind(email, new Date().toISOString(), new Date().toISOString()).run();
      return Response.json({ success: true, message: "Consent recorded – chatbot unlocked!" });
    }

    if (!sessionId) return Response.json({ error: "sessionId required" }, { status: 400 });

    // Load conversation history from KV
    let history = await env.KV.get(`chat:${sessionId}`, { type: "json" }) || [];

    if (action === "send-message") {
      history.push({ role: "user", content: message });

      const aiResponse = await env.AI.run("@cf/meta/llama-3.1-8b-instruct-fast", {
        messages: [
          {
            role: "system",
            content: `You are a friendly DeFi Risk Advisor Chatbot. Ask 1 question at a time to determine the user's risk profile.
            Topics to cover: investment goals, DeFi experience, loss tolerance, portfolio size, preferred chains, time horizon.
            After 5-7 questions, when user says "finish" or "report", output ONLY this JSON:
            {
              "riskProfile": "Conservative / Moderate / Aggressive",
              "summary": "2-3 paragraph personalized summary",
              "recommendations": [ { "project": "...", "reason": "...", "riskLevel": "Low/Medium/High" } ],
              "pdfTitle": "Your Personalized DeFi Risk Profile Report"
            }
            Be conversational and helpful until then.`
          },
          ...history
        ],
        max_tokens: 1200,
        temperature: 0.7
      });

      const responseText = aiResponse.response || aiResponse;
      history.push({ role: "assistant", content: responseText });

      // Save history
      await env.KV.put(`chat:${sessionId}`, JSON.stringify(history), { expirationTtl: 3600 });

      // Check if final report
      let finalReport = null;
      try {
        if (responseText.includes('{') && responseText.includes('riskProfile')) {
          finalReport = JSON.parse(responseText.match(/\{[\s\S]*\}/)[0]);
        }
      } catch (e) {}

      return Response.json({
        success: true,
        reply: responseText,
        finalReport
      });
    }

    return Response.json({ error: "invalid action" }, { status: 400 });
  }
};
```

### Prompt 5: Add consent + chatbot UI to Jekyll (main Repl)
```
In your main Jekyll Repl, create assets/js/risk-chatbot.js with this full frontend code (chat UI + jsPDF + email):

// assets/js/risk-chatbot.js
let sessionId = 'session-' + Date.now();

async function startRiskChatbot() {
  const email = document.getElementById('lead-email').value.trim();
  if (!email) return alert("Please enter your email");

  // Submit consent & lead
  await fetch('https://defiscoring-risk-chatbot.YOURUSERNAME.workers.dev', {
    method: 'POST',
    body: JSON.stringify({ action: 'submit-consent', email, consent: true })
  });

  // Show chat UI
  document.getElementById('chatbot-container').classList.remove('hidden');
  document.getElementById('consent-form').classList.add('hidden');

  addMessage("assistant", "Hi! I'm your DeFi Risk Advisor. To give you the best recommendations, I'll ask a few questions about your goals and experience. Ready?");
}

function addMessage(role, text) {
  const chat = document.getElementById('chat-messages');
  const div = document.createElement('div');
  div.className = `mb-4 ${role === 'user' ? 'text-right' : 'text-left'}`;
  div.innerHTML = `<div class="inline-block max-w-[80%] px-5 py-3 rounded-3xl ${role === 'user' ? 'bg-blue-600 text-white' : 'bg-gray-200 dark:bg-gray-800'}">${text}</div>`;
  chat.appendChild(div);
  chat.scrollTop = chat.scrollHeight;
}

async function sendChatMessage() {
  const input = document.getElementById('chat-input');
  const msg = input.value.trim();
  if (!msg) return;
  addMessage('user', msg);
  input.value = '';

  const res = await fetch('https://defiscoring-risk-chatbot.YOURUSERNAME.workers.dev', {
    method: 'POST',
    body: JSON.stringify({ action: 'send-message', sessionId, message: msg })
  });
  const data = await res.json();

  addMessage('assistant', data.reply);

  if (data.finalReport) {
    generateAndEmailPDF(data.finalReport);
  }
}

async function generateAndEmailPDF(report) {
  // jsPDF CDN already loaded
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF();
  doc.setFontSize(20);
  doc.text(report.pdfTitle || "Your DeFi Risk Profile Report", 20, 30);
  doc.setFontSize(12);
  doc.text(`Risk Profile: ${report.riskProfile}`, 20, 50);
  doc.text(report.summary, 20, 70, { maxWidth: 160 });
  // Add recommendations...
  doc.save(`DeFi-Risk-Report-${new Date().toISOString().slice(0,10)}.pdf`);

  // Optional: send email via another Worker or alert user
  alert("✅ PDF downloaded! We can also email it to you if you confirm.");
}
```
